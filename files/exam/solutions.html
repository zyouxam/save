小猫爬山

这一题如果我们用递归枚举一只猫坐上的缆车，我们可以得出，倘若第i只猫坐在第j辆缆车上，那么一定有
那么，代码就分为以下几个步骤：
1. 首先从小到大枚举要用的缆车数量,for i=1 to n。
2. 递归枚举每只猫所乘坐的缆车，参数x为当前猫的序号，for o=1~min(x,i)，用一个f数组表示第i个缆车所乘猫的总重量，倘若，那么递归下去。
3. 倘若枚举完了所有猫，意味着当前所用的缆车个数符合条件，直接输出。

  6. struct link

  11. link e[MAXN << 1];

  13. int n, u, v, dp[MAXN][2], happy[MAXN];
  15. inline int mymax(int a, int b)
  16. {
  17.     return a > b ? a : b;
  18. }
  20. inline void add(int u, int v)
  21. {
  22.     e[cnt] = (link)
  23.     {
  24.         v, head[u]
  25.     };
  26.     head[u] = cnt++;
  27.     e[cnt] = (link)
  28.     {
  29.         u, head[v]
  30.     };

  32. }
  34. inline void TreeDP(int x, int father)
  35. {
  36.     for (int i = head[x]; ~i; i = e[i].nxt)
  37.         if (e[i].to != father)
  38.         {
  39.             TreeDP(e[i].to, x);
  40.             dp[x][0] += mymax(dp[e[i].to][1], dp[e[i].to][0]);
  41.             dp[x][1] += dp[e[i].to][0];
  42.         }

  44.     return ;
  45. }
  47. inline int read()
  48. {
  49.     int x = 0;

  53.     {
  54.         if (ch == '-') f = true;
  55.         ch = getchar();
  56.     }

  61. int main()

小烈送菜

老姚版题解

  1. #include <bits/stdc++.h> 
  2. const int maxn=2500+5;
  3. int a[maxn],f[maxn][maxn];
  4. void Solve(){
  5.     int n , ans = 0;
  6.     scanf("%d",&n); 
  7.     for(int i = 1; i <= n; i++)
  8.         scanf("%d",&a[i]);
  9.     for(int i = 1; i <= n; i++){//前面的小烈
  10.         for(int j = 0; j < i; j++){//后面的小烈 
  11.             f[i+1][j] = std::max(f[i+1][j], f[i][j] + a[i] * a[i+1]);
  12.             f[i+1][i] = std::max(f[i+1][i], f[i][j] + a[j] * a[i+1]);
  13.         }
  14.     }
  15.     for(int i = 0; i < n; i++)
  16.         ans  = std::max(ans, f[n][i] + a[n] * a[i]);
  17.         //这里,小烈a已经走到头,小烈b可能在0到n-1之间的任何位置
  18.         //两个小烈必须汇合才能表示走完了一圈,所以要加上a[n] * a[i] 
  19.     printf("%d\n",ans); 
  20. }
  21. int main(){
  22.     Solve();
  23.     return 0;
  24. }

这Seg确实不好写，不过因为它与ai的相对顺序无关，所以，我们在对ai排序之后，就可做了。维护一个区间最大值，维护一个和，维护一个区间赋值的懒标记，再维护一个时间变化的标记就可以了。

因为不论怎样在排序过后的序列里面右边的一定不小于左边的，所以我们可以在线段树里面二分来找到每一次操作的开始位置，然后就可以区间修改了。

  1. #include<cstdio>
  2. #include<algorithm>
  3. #define N 1050000
  4. typedef long long ll;
  5. int n,m,i,a[N/2];ll now,old,R,ans;
  6. inline void read(int&a){char c;while(!(((c=getchar())>='0')&&(c<='9')));a=c-'0';while(((c=getchar())>='0')&&(c<='9'))(a*=10)+=c-'0';}
  7. inline void read(ll&a){char c;while(!(((c=getchar())>='0')&&(c<='9')));a=c-'0';while(((c=getchar())>='0')&&(c<='9'))(a*=10LL)+=c-'0';}
  8. struct P{
  9.   bool a;ll b,c;
  10.   P(){a=1,b=c=0;}
  11.   P(bool _a,ll _b,ll _c){a=_a,b=_b,c=_c;}
  12.   P operator+(P B){return P(a&B.a,b*B.a+B.b,c*B.a+B.c);}
  13. }tag[N];
  14. struct Node{ll vl,vr,vs,s;}T[N];
  15. inline void add(int x,P p,int a,int b){
  16.   if(p.a){
  17.     T[x].vl+=p.b+p.c*::a[a];
  18.     T[x].vr+=p.b+p.c*::a[b];
  19.     T[x].vs+=p.b*(b-a+1)+p.c*T[x].s;
  20.   }else{
  21.     T[x].vl=p.b+p.c*::a[a];
  22.     T[x].vr=p.b+p.c*::a[b];
  23.     T[x].vs=p.b*(b-a+1)+p.c*T[x].s;
  24.   }
  25.   tag[x]=tag[x]+p;
  26. }
  27. inline void pb(int x,int a,int b){
  28.   int mid=(a+b)>>1;
  29.   add(x<<1,tag[x],a,mid),add(x<<1|1,tag[x],mid+1,b);
  30.   tag[x]=P();
  31. }
  32. void build(int x,int a,int b){
  33.   if(a==b){T[x].s=::a[a];return;}
  34.   int mid=(a+b)>>1;
  35.   build(x<<1,a,mid),build(x<<1|1,mid+1,b);
  36.   T[x].s=T[x<<1].s+T[x<<1|1].s;
  37. }
  38. void check(int x,int a,int b){
  39.   if(T[x].vr<=R)return;
  40.   if(T[x].vl>R){
  41.     ans+=T[x].vs-R*(b-a+1);
  42.     add(x,P(0,R,0),a,b);
  43.     return;
  44.   }
  45.   pb(x,a,b);
  46.   int mid=(a+b)>>1;
  47.   check(x<<1,a,mid),check(x<<1|1,mid+1,b);
  48.   T[x].vl=T[x<<1].vl,T[x].vr=T[x<<1|1].vr;
  49.   T[x].vs=T[x<<1].vs+T[x<<1|1].vs;
  50. }
  51. int main(){
  52.   for(read(n),read(m),i=1;i<=n;i++)read(a[i]);
  53.   std::sort(a+1,a+n+1);
  54.   build(1,1,n);
  55.   while(m--){
  56.     read(now),read(R);
  57.     add(1,P(1,0,now-old),1,n);
  58.     ans=0;
  59.     check(1,1,n);
  60.     old=now;
  61.     printf("%lld\n",ans);
  62.   }
  63.   return 0;
  64. }
