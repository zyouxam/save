#include <cstdio>
#include <cstring>
#include <queue>
using namespace std;
const int N = 100010, M = 300010;
long long dis[N], ans;
int head[N], tot, n, m, cnt[N], op, a, b;
bool vis[N];
struct Edge { int to, next, w; } edges[M];
void add(int x, int y, int w) { edges[++tot] = (Edge){y, head[x], w}, head[x] = tot; }
bool spfa() {
    memset(dis, -1, sizeof(dis));
    queue<int> q;
    dis[0] = 0;
    for (int i = 0; i <= n; i++) q.push(i), vis[i] = cnt[i] = 1;
    int c = 0;
    while (q.size()) {
        int x = q.front();
        q.pop();
        vis[x] = 0;
        for (int i = head[x]; i; i = edges[i].next) {
            int y = edges[i].to;
            if (dis[y] < dis[x] + edges[i].w) {
                if (++c > 2*N) return 1;
                cnt[y] = cnt[x] + 1;
                dis[y] = dis[x] + edges[i].w;
                if (cnt[y] > n + 1) return 1;
                if (!vis[y]) q.push(y), vis[y] = 1;
            }
        }
    }
    return 0;
}
int main() {
    freopen("bomb.in", "r", stdin);
    freopen("bomb.out", "w", stdout);
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; i++) {
        scanf("%d%d%d", &op, &a, &b);
        if (op == 1) add(a, b, 0), add(b, a, 0);
        else if (op == 2) add(a, b, 1);
        else if (op == 3) add(b, a, 0);
        else if (op == 4) add(b, a, 1);
        else add(a, b, 0);
    }
    for (int i = 1; i <= n; i++) add(0, i, 1);
    if (spfa()) puts("-1");
    else {
        for (int i = 1; i <= n; i++) ans += dis[i];
        printf("%lld", ans);
    }
    return 0;
}
