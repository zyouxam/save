#include <cstdio>
#include <cstring>
const int N = 305;
int f[N][N], n;
char s[N];
int dfs(int l, int r) {
    if (l > r) return f[l][r] = 0;
    if (l == r) return f[l][r] = 1;
    if (f[l][r] != -1) return f[l][r];
    f[l][r] = 0;
    for (int k = l + 2; k <= r; k++)
        if (s[l + 1] == s[k - 1] && s[k] == s[r])
            f[l][r] = (f[l][r] + (long long)dfs(l + 1, k - 1) * dfs(k, r)) % (int)1e9;
    return f[l][r];
}
int main() {
    freopen("school.in", "r", stdin);
    freopen("school.out", "w", stdout);
    memset(f, -1, sizeof(f));
    scanf("%s", s + 1);
    printf("%d\n", dfs(1, n = strlen(s + 1)));
}
