#include <cstdio>
const int MAXN=100010;
struct edge{
    int to,next;
} edges[MAXN*2];
char opt;
int f[MAXN],fa[MAXN],head[MAXN*2],tot=0,black[MAXN];
void add(int x,int y) {
    edges[++tot].to=y;
    edges[tot].next=head[x];
    head[x]=tot;
}
int find(int x){return x==f[x]?x:(f[x]=find(f[x]));}
void join(int a,int b){f[find(a)]=find(b);}
void dfs(int x,int father) {
    if(black[x]) f[x]=x;
    else f[x]=father;
    fa[x]=father;
    for(int i=head[x];i;i=edges[i].next) {
        int v=edges[i].to;
        if(v==father) continue;
        dfs(v,x);
    }
}
int que[MAXN][3];
int main() {
    freopen("yuan.in","r",stdin);
    freopen("yuan.out","w",stdout);
    int n,m;
    scanf("%d%d",&n,&m);
    int a,b;
    for(int i=1;i<n;++i) {
        scanf("%d%d",&a,&b);
        add(a,b);
        add(b,a);
    }
    black[1]=true;
    for(int i=1;i<=m;++i) {
        scanf(" %c%d",&opt,&que[i][1]);
        if(opt=='Q'){
            que[i][0]=0;
        }else{
            que[i][0]=1;
            black[que[i][1]]++;
        }
    }
    dfs(1,0);
    fa[1]=1;
    for(int i=m;i>=1;--i) {
        if(que[i][0]) {
            black[que[i][1]]--;
            if(!black[que[i][1]]) f[que[i][1]]=fa[que[i][1]];
        } else { 
            que[i][2]=find(que[i][1]);
        }
    }
    for(int i=1;i<=m;++i) {
        if(!que[i][0]) {
            printf("%d\n",que[i][2]);
        }
    }
    return 0;
}
