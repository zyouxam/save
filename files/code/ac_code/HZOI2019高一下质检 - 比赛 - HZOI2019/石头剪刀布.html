#include <cstdio>
int fa[150005];
int n, k, ans;
int find(int x) { return x == fa[x] ? x : (fa[x] = find(fa[x])); }
int join(int x, int y) { fa[find(x)] = find(y); }
int main() {
    freopen("sjb.in","r",stdin);
    freopen("sjb.out","w",stdout);
    int x, y, z;
    scanf("%d%d", &n, &k);
    for (int i = 1; i <= 3 * n; ++i)
        fa[i] = i;
    for (int i = 1; i <= k; ++i) {
        scanf("%d%d%d", &z, &x, &y);
        if (x > n || y > n) {
            ans++;
            continue;
        }
        if (z == 1) {
            if (find(x + n) == find(y) || find(x + 2 * n) == find(y)) {
                ans++;
                continue;
            }
            join(x, y);
            join(x + n, y + n);
            join(x + 2 * n, y + 2 * n);
        } else if (z == 2) {
            if (x == y) {
                ans++;
                continue;
            }
            if (find(x) == find(y) || find(x + 2 * n) == find(y)) {
                ans++;
                continue;
            }
            join(x, y + 2 * n);
            join(x + n, y);
            join(x + 2 * n, y + n);
        }
    }
    printf("%d\n", ans);
    return 0;
}
