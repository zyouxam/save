#include<cstdio>
#include <algorithm>
using namespace std;
#define P pair<int, int>
#define maxe first
#define mine second
const int N = 50005;
int data[N];
P tree[N*4];
int n;
void build(int root, int l, int r){
    if (l==r){
        tree[root].maxe = tree[root].mine = data[l];
        return;
    } else if (l>r)return;
    int mid= (l + r) >> 1;
    build(root<<1,l,mid);
    build(root<<1|1,mid+1,r);
    tree[root].maxe = max(tree[root<<1].maxe,tree[root<<1|1].maxe);
    tree[root].mine = min(tree[root<<1].mine,tree[root<<1|1].mine);
}
P query(int root,int l, int r, int s, int e){
    if(l>r)return make_pair(0,0);
    else if(l==r)return tree[root];
    int mid= (l + r) >> 1;
    if(s<=l && e >= r)return tree[root];
    else if(e<=mid)return  query(root<<1,l,mid,s,e);
    else if(s>mid)return query(root<<1|1,mid+1,r,s,e);
    P nl=query(root<<1,l,mid,s,e),nr=query(root<<1|1,mid+1,r,s,e);
    return make_pair(max(nl.maxe,nr.maxe),min(nl.mine,nr.mine));
}
int main() {
    freopen("lineup.in","r",stdin);
    freopen("lineup.out","w",stdout);
    int q,a,b;
    scanf("%d%d",&n,&q);
    for(int i=1;i<=n; i++)scanf("%d", &data[i]);
    build(1,1,n);
    while(q--){
        scanf("%d%d",&a, &b);
        P p = query(1,1,n,a,b);
        printf("%d\n",p.maxe-p.mine);
    }
    return 0;
}
