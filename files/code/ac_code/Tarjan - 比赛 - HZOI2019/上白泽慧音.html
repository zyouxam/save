#include<cstdio>
#include<algorithm>
#define MAXM 50005
#define MAXN 10005
using namespace std;
struct edge{
    int f,to,next;
    edge(){to=next=0;}
} edges[MAXM];
int head[MAXM],tot=1;
void add(int x,int y){
    edges[++tot].to=y;
    edges[tot].f=x;
    edges[tot].next=head[x];
    head[x]=tot;
}
int n,m,x,y,z;
int dfn[MAXN],low[MAXN],cnt,dfs_clock,sta[MAXN],top,belong[MAXN],scc_cnt,siz[MAXN];
void tarjan(int u){
    
    dfn[u] = low[u] = ++ dfs_clock;
    sta[++top] = u; 
    
    for(int x = head[u];x;x=edges[x].next){
        int v=edges[x].to;
        
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u], low[v]);
        }else if(!belong[v]) low[u] = min(low[u],dfn[v]);
    }
    if(dfn[u] == low[u]){
        scc_cnt++;
        while(1){
            int x = sta[top--]; 
            belong[x] = scc_cnt; 
            siz[scc_cnt] ++ ; 
            if(x == u) break; 
        }
    }
}
int out[MAXN];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++){
        scanf("%d%d%d",&x,&y,&z);
        add(x,y);
        if(z==2)add(y,x);
    }
    for(int i=1;i<=n;i++)if(!dfn[i])tarjan(i);
    
    int mi=max_element(siz+1,siz+scc_cnt+1)-siz;
    printf("%d\n",siz[mi]);
    for(int i=1;i<=n;i++){
        if(belong[i]==mi){
            printf("%d ",i);
        }
    }
    return 0;
}
