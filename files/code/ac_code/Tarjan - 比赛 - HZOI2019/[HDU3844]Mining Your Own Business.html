#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>
typedef long long ll;
const int MAXN = 100100;
const int MAXM = 100010;
using namespace std;
int low[MAXN], dfn[MAXN], Stack[MAXN];
int dfs_clock, top, block;
bool iscut[MAXN];
vector<int> bcc[MAXN];
struct Edge {
    int to, next;
} edge[MAXM << 1];
int head[MAXN], tot;
void addedge(int u, int v) {
    edge[tot].to = v;
    edge[tot].next = head[u];
    head[u] = tot++;
}
void init() {
    tot = 0;
    memset(head, 0xff, sizeof(head));
}
void Tarjan(int u, int pre) {
    int v;
    low[u] = dfn[u] = ++dfs_clock;
    Stack[top++] = u;
    int son = 0;
    for (int i = head[u]; i != -1; i = edge[i].next) {
        v = edge[i].to;
        if (v == pre) continue;
        if (!dfn[v]) {
            son++;
            Tarjan(v, u);
            low[u] = min(low[u], low[v]);
            if (low[v] >= dfn[u]) {
                iscut[u] = 1;
                block++;
                bcc[block].clear();
                int vn;
                for (;;) {
                    vn = Stack[--top];
                    bcc[block].push_back(vn);
                    if (vn == v) break;
                }
                bcc[block].push_back(u);
            }
        } else if (low[u] > dfn[v]) {
            low[u] = dfn[v];
        }
    }
    if (pre < 0 && son == 1) iscut[u] = false;
}
void solve(int n) {
    memset(dfn, 0, sizeof(dfn));
    memset(iscut, false, sizeof(iscut));
    dfs_clock = top = block = 0;
    for (int i = 1; i <= n; ++i)
        if (!dfn[i]) Tarjan(i, -1);
    ll ans1 = 0, ans2 = 1;
    if (block == 1) {
        ans1 = 2ll;
        ans2 = (ll)n * (n - 1) / 2;
    } else {
        for (int i = 1; i <= block; ++i) {
            int z = bcc[i].size();
            int cut_cnt = 0;
            for (int j = 0; j < z; ++j) {
                if (iscut[bcc[i][j]]) cut_cnt++;
            }
            if (cut_cnt == 1) {
                ans1++;
                ans2 *= (ll)(z - 1);
            }
        }
    }
    printf("%lld %lld\n", ans1, ans2);
}
int main() {
    int m, u, v, cas = 1;
    while (scanf("%d", &m) && m) {
        int n = 0;
        init();
        while (m--) {
            scanf("%d%d", &u, &v);
            addedge(u, v);
            addedge(v, u);
            n = max(n, max(u, v));
        }
        printf("Case %d: ", cas++);
        solve(n);
    }
    return 0;
}
