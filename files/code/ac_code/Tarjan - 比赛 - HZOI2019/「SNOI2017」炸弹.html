#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cmath>
#include<ctime>
using namespace std;
typedef long long LL;
const LL mod=1e9+7;
struct node{LL r,x;}w[510000];
int n,p[510000];
int lef[510000],rig[510000];
LL minL[510000],maxR[510000];
void pre(int u)
{
    while(1)
    {
        bool bk=false;
        while(lef[u]>1 && minL[u]<=w[lef[u]-1].x)
        {
            minL[u]=min(minL[u],minL[lef[u]-1]);maxR[u]=max(maxR[u],maxR[lef[u]-1]);
            rig[u]=max(rig[u],rig[lef[u]-1]);lef[u]=min(lef[u],lef[lef[u]-1]);
            bk=true;
        }
        while(rig[u]<n && maxR[u]>=w[rig[u]+1].x)
        {
            minL[u]=min(minL[u],minL[rig[u]+1]);maxR[u]=max(maxR[u],maxR[rig[u]+1]);
            lef[u]=min(lef[u],lef[rig[u]+1]);rig[u]=max(rig[u],rig[rig[u]+1]);
            bk=true;
        }
        if(bk==false)return ;
    }
}
int main()
{
    srand(time(0));
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%lld%lld",&w[i].x,&w[i].r);
    for(int i=1;i<=n;i++)lef[i]=rig[i]=i,minL[i]=w[i].x-w[i].r,maxR[i]=w[i].x+w[i].r;
    for(int i=1;i<=n;i++)p[i]=i;
    random_shuffle(p+1,p+1+n);
    for(int i=1;i<=n;i++)pre(p[i]);
    LL ans=0;
    for(LL i=1;i<=n;i++)ans=(ans+(LL)(i*(rig[i]-lef[i]+1))%mod)%mod;
    printf("%lld\n",ans);
    return 0;
}
