#include<bits/stdc++.h>
using namespace std;
#define int long long
int euler(int a){
	int ans=a;
	for(int i=2;i*i<=a;++i)
		if(a%i==0){
			ans=ans/i*(i-1);
			while(a%i==0)a/=i;
		}
	if(a!=1)ans=ans/a*(a-1);
	return ans;
}
int n,ans;
signed main(){
	scanf("%lld",&n);
	for(int i=1;i*i<=n;++i)
		if(n%i==0){
			ans+=i*euler(n/i);
			if(i!=n/i) ans+=(n/i)*euler(i);
		}
	printf("%lld",ans);
	return 0;
}
