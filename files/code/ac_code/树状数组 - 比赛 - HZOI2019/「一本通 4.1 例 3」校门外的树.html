#include <cstdio>
#include <cmath>
#include <cstring>
#define N 100005
#define lowbit(x) ((x)&(-(x)))
using namespace std;
int n,m,h[N],t[N];
void add(int a[],int k){
    for(int i=k;i<=n;i+=lowbit(i)) a[i]++;
}
int sum(int a[],int k){
    int ans=0;
    for(int i=k;i>0;i-=lowbit(i)) ans+=a[i];
    return ans;
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++){
        int k,x,y;
        scanf("%d%d%d",&k,&x,&y);
        if(k==1){
            add(h,x);
            add(t,y);
        }
        else{
            printf("%d\n",sum(h,y)-sum(t,x-1));
        }
    }
    return 0;
}
