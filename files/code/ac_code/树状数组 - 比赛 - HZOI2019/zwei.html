#include<cstring>
#include<cstdio>
using namespace std;
const int N=10e5+5;
const int M=10e5+5;
//const int N=10;
//const int M=10;
int n,m,c[M],t,a[M];
void update(int i,int d){
    //int p=c[i];
    //c[i]^=d;
    //i+=i&(-i);
    int l=i;
    while(i<=n){
        c[i]^=a[l];
        c[i]^=d;
        i+=i&(-i);
    }
}
int ask(int i){
    int ans=0;
    while(i>0){
        ans^=c[i];
        i-=i&(-i);
    }
    return ans;
}
int main(){
    scanf("%d%d",&n,&m);
    n;
    for(int i=1;i<=n;i++){
        scanf("%d",&t);
        update(i,t);
        a[i]=t;
    }
    int opt,x,y;
    for(int i=1;i<=m;i++){
        scanf("%d%d%d",&opt,&x,&y);
        if(opt==0){
            update(x,y);
            a[x]=y;
        }
        else printf("%d\n",ask(x-1)^ask(y));
    }
    return 0;
}
