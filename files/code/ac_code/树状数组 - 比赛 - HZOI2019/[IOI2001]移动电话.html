#include<cstring>
#include<cstdio>
using namespace std;
const int maxn=1030;
int c[maxn][maxn]={0},n;
inline int lowbit(int x){
    return x&-x;
}
int query(int x,int y){
    int tot=0;
    for(int i=x;i>0;i-=lowbit(i))
        for(int j=y;j>0;j-=lowbit(j))
            tot+=c[i][j];
    return tot;
}
void update(int x,int y,int z){
    for(int i=x;i<=n;i+=lowbit(i))
        for(int j=y;j<=n;j+=lowbit(j))
            c[i][j]+=z;
}
int main(){
    int flag;
    while(scanf("%d",&flag),flag!=3){
        if(flag==0){
            scanf("%d",&n);continue;
        }
        if(flag==1){
            int x,y,z;scanf("%d%d%d",&x,&y,&z);
            x++;y++;
            update(x,y,z);
        }
        if(flag==2){
            int x1,y1,x2,y2;scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
            x1++;x2++;y1++;y2++;
            int k=query(x2,y2)+query(x1-1,y1-1)-query(x2,y1-1)-query(x1-1,y2);
            printf("%d\n",k);
        }
    }
    return 0;
}
