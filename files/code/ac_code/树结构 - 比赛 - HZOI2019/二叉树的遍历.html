#include<cstdio>
#include<iostream>
using namespace std;
int n,tree[30][2],father[30];
char data[30];
int find(int x){
	if(father[x]==0)return x;
	return find(father[x]);
}
int f(int x){
	printf("%c",data[x]);
	if(tree[x][0])f(tree[x][0]);
	if(tree[x][1])f(tree[x][1]);
}
int m(int x){
	if(tree[x][0])m(tree[x][0]);
	printf("%c",data[x]);
	if(tree[x][1])m(tree[x][1]);
}
int b(int x){
	if(tree[x][0])b(tree[x][0]);
	if(tree[x][1])b(tree[x][1]);
	printf("%c",data[x]);
}
int main(){
	freopen("bianli.in","r",stdin);
	freopen("bianli.out","w",stdout);
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		getchar();
		scanf("%c%d%d",&data[i],&tree[i][0],&tree[i][1]);
		father[tree[i][0]]=father[tree[i][1]]=i;
	}
	int h=find(1);
	f(h);printf("\n");
	m(h);printf("\n");
	b(h);printf("\n");
}
