#include<cstdio>
#include<bitset>
using namespace std;
bitset<3000000> tree;
int d,bi;
int go(int root,int step){
	if(step>=d)return root;
	else if(tree[root]){
		tree[root]=0;
		return go(root*2+1,step+1);
	} else {
		tree[root]=1;
		return go(root*2,step+1);
	}
}
int main(){
	freopen("drop.in","r",stdin);
	freopen("drop.out","w",stdout);
	scanf("%d%d",&d,&bi);
	int bn;
	for(int i=0;i<bi;i++)bn=go(1,1);
	printf("%d",bn);
	return 0;
}
