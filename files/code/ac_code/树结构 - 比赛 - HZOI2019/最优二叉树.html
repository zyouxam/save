#include<cstdio>
#include<queue>
using namespace std;
int n;
unsigned long long input[55];

int main(){
	freopen("best.in","r",stdin);
	freopen("best.out","w",stdout);
	scanf("%d",&n);
	priority_queue<unsigned long long,vector<unsigned long long>,greater<unsigned long long> > min_q;
	priority_queue<unsigned long long> max_q;
	for(int i=0;i<n;i++){
		scanf("%llu",&input[i]);
		min_q.push(input[i]);
		max_q.push(input[i]);
	}
	for(int i=1;i<n;i++){
		unsigned long long a,b;
		a=min_q.top();min_q.pop();
		b=min_q.top();min_q.pop();
		//printf(">>%d >>%d <<%d\n",a,b,a*b+1);
		min_q.push(a*b+1);
		
		a=max_q.top();max_q.pop();
		b=max_q.top();max_q.pop();
		//printf(">>%d >>%d <<%d\n",a,b,a*b+1);
		max_q.push(a*b+1);
	}
	printf("%llu\n%llu\n%llu\n",min_q.top(),max_q.top(),min_q.top()-max_q.top());
	return 0;
}
