#include<cstdio>
#include<vector>
#include<iostream>
using namespace std;
int n,m;
int tree[205];
vector<int> child[105];
int find(int x){
	if(tree[x]==0)return x;
	return find(tree[x]);
}
int main(){
	freopen("find.in","r",stdin);
	freopen("find.out","w",stdout);
	int mc=-1,mi=0;
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		tree[y]=x;
		child[x].push_back(y);
		//printf("%d %d %d %d %d\n",mi,x,y,mc,int(child[x].size()));
		if(mc<int(child[x].size()) || mc==int(child[x].size()) && x>mi){
			mc=int(child[x].size());
			mi=x;
		}
	}
	printf("%d\n%d\n",find(1),mi);
	for(int i=0;i<child[mi].size();i++){
		printf("%d ",child[mi][i]);
	}
	
}
