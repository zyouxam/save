#include<cstdio>
#include<string>
#include<iostream>
#define CUT(x,a,b) (x.substr(a,b-a))
using namespace std;
void f(string ms,string cs){
	if(ms=="")return;
	if(ms.size()==1){
		cout<<ms;
	}else{
		string root=CUT(cs,0,1);
		int ri=ms.find(root,0);
		string left_cs,right_cs,left_ms=CUT(ms,0,ri),right_ms=CUT(ms,ri+1,ms.size());
		bool hash_left[128]={},hash_right[128]={};
		for(int i=0;i<ri;i++)hash_left[int(ms[i])]=1;
		for(int i=ri+1;i<int(ms.size());i++)hash_right[int(ms[i])]=1;
		for(int i=0;i<int(cs.size());i++){
			if(hash_left[int(cs[i])])left_cs+=cs[i];
			if(hash_right[int(cs[i])])right_cs+=cs[i];
		}
		//cout<<"\nms:"<<ms<<" cs:"<<cs<<" left_ms:"<<left_ms<<" right_ms:"<<right_ms<<" left_cs:"<<left_cs<<" right_cs:"<<right_cs<<endl;
		cout<<root;
		f(left_ms,left_cs);
		f(right_ms,right_cs);
	}
}
int main(){
	freopen("flist.in","r",stdin);
	freopen("flist.out","w",stdout);
	string ms,cs;
	cin>>ms>>cs;
	f(ms,cs);
	return 0;
}
