#include<cstdio>
#include<string>
#include<iostream>
#define slice(x,a,b) (x.substr(a,b-a+1))
using namespace std;
void f(string fs,string ms){
	string root=slice(fs,0,0);
	int ri=ms.find(root,0);
	if(fs=="")return;
	if(fs.size()==1){
		cout<<fs;
	}else{
		f(slice(fs,1,ri),slice(ms,0,ri-1));
		f(slice(fs,ri+1,fs.size()-1),slice(ms,ri+1,ms.size()-1));
		cout<<root;
	}
}
int main(){
	freopen("post.in","r",stdin);
	freopen("post.out","w",stdout);
	string fs,ms;
	cin>>fs>>ms;
	f(fs,ms);
	return 0;
}
