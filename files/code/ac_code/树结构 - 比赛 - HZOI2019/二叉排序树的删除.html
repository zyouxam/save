#include<cstdio>
#include<iostream>
using namespace std;
struct node{
	int data;
	bool have_data;
	node *l,*r,*p;
	node(){
		l=r=p=NULL;
		have_data=false;
	}
};
void insert(node* root,int n,node* p){
	if(!root->have_data){
		root->data=n;
		root->p=p;
		root->have_data=true;
		return;	
	} else if (root->data<n){
		if(!(root->r))root->r=new node;
		insert(root->r,n,root);
	} else {
		if(!(root->l))root->l=new node;
		insert(root->l,n,root);
	}
}
node* get_max(node* root){
	node* i;
	for(i=root;i->r;i=i->r);
	return i;
}
node* del(node* root,int key){
	if(root->data==key){
		if(root->l && root->r){
			node* target=get_max(root->l);
			root->data=target->data;
			delete target;
			return root;
		}else if(root->l || root->r){
			node *target=(root->l)?(root->l):(root->r);
			delete root;
			return target;
		}else{
			delete root;
			return NULL;
		}
	}else if(root->data<key){
		if(root->r)root->r=del(root->r,key);
		else return root;
		return root;
	}else{
		if(root->l)root->l=del(root->l,key);
		else return root;
		return root;
	}
}
void print(node* root){
	if(!(root))return;
	if(root->have_data)printf("%d ",root->data);
	if(root->l)print(root->l);
	if(root->r)print(root->r);
}
void mprint(int a){
	printf("%d ",a);
}
int main(){
	freopen("del.in","r",stdin);
	freopen("del.out","w",stdout);
	int n;
	scanf("%d",&n);
	node* root=new node;
	for(int i=0;i<n;i++){
		int c;
		scanf("%d",&c);
		insert(root,c,NULL);
	}
	int dn,dt;
	scanf("%d",&dn);
	for(int i=0;i<dn;i++){
		scanf("%d",&dt);
		root=del(root,dt);
		print(root);
		printf("\n");
	}
	return 0;
}
