#include<cstdio>
#include<string>
#include<iostream>
#define CUT(x,a,b) (x.substr(a,b-a))
using namespace std;
int count[128];
int f(string fs,string ms){
	char root=fs[0];
	int ri=ms.find(root,0);
	if(fs=="")return 0;
	if(fs.size()==1){
		return count[int(root)]=1;
	}else{
		return count[int(root)]=f(CUT(fs,1,ri+1),CUT(ms,0,ri+1))+f(CUT(fs,ri+1,fs.size()),CUT(ms,ri+1,ms.size()));
	}
}
int main(){
	freopen("btout.in","r",stdin);
	freopen("btout.out","w",stdout);
	string fs,ms;
	cin>>fs>>ms;
	count[int(fs[0])]=f(fs,ms);
	for(int i=0;i<int(fs.size());i++){
		for(int j=0;j<count[int(fs[i])];j++){
			cout<<fs[i];
		}
		cout<<endl;
	}
	return 0;
}
