#include<iostream>
#include<string>
using namespace std;
int noden=1;
struct node{
	node * c[26];
	node(){
		for(int i=0;i<26;i++)c[i]=NULL;
	}
};
int main(){
	freopen("wordtree.in","r",stdin);
	freopen("wordtree.out","w",stdout);
	string l;
	node * head=new node;
	while(cin>>l){
		node * now=head;
		for(int i=0;i<l.size();i++){
			int ci=l[i]-'A';
			if(now->c[ci]==NULL){
				node* n=new node;
				now->c[ci]=n;
				noden++;
			}
			now=now->c[ci];
		}
	}
	cout<<noden;
}
