#include<cstdio>
#include<iostream>
using namespace std;
int tree[101][3];
int mi	,key,n;
int f(int root){
	if(tree[root][1] && f(tree[root][1]))return 1;
	mi++;
	if(tree[root][0]==key)return 1;
	if(tree[root][2] && f(tree[root][2]))return 1;
	return 0;
}
int main(){
	freopen("treea.in","r",stdin);
	freopen("treea.out","w",stdout);
	scanf("%d%d",&n,&key);
	for(int i=1;i<=n;i++){
		scanf("%d%d%d",&tree[i][0],&tree[i][1],&tree[i][2]);
	}
	f(1);
	printf("%d",mi);
	return 0;
}
