#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
string fs;
int tail(int root){
	if(fs[root]=='.')return root;
	else return tail(tail(root+1)+1);
}
void m(int root){
	int l=tail(root);
	if(l-root==0)return;
	else if(l-root==2)cout<<fs[root];
	else{
		m(root+1);//left
		cout<<fs[root];
		m(tail(root+1)+1);//right
	}
}
void b(int root){
	int l=tail(root);
	if(l-root==0)return;
	else if(l-root==2)cout<<fs[root];
	else{
		b(root+1);//left
		b(tail(root+1)+1);//right
		cout<<fs[root];
	}
}
int main(){
	freopen("expand.in","r",stdin);
	freopen("expand.out","w",stdout);
	cin>>fs;
	m(0);cout<<endl;
	b(0);cout<<endl;
	return 0;
}
