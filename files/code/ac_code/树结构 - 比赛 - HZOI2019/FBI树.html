#include<cstdio>
#include<string>
#include<iostream>
#define CUT(x,a,b) (x.substr(a,b-a))
using namespace std;
char judge(string s){
	bool B=(s[0]=='0'?true:false),I=(s[0]=='1'?true:false);
	for(int i=0;i<int(s.size());i++){
		if(s[i]=='1' && B)return 'F';
		else if(s[i]=='0' && I)return 'F';
	}
	return B?'B':'I';
}
void f(string s){
	int l=s.size();
	if(l!=1){
		f(CUT(s,0,l/2));
		f(CUT(s,l/2,l));
	}
	cout<<judge(s);
}
int main(){
	freopen("fbi.in","r",stdin);
	freopen("fbi.out","w",stdout);
	int n;
	string s;
	cin>>n>>s;
	f(s);
	return 0;
}
