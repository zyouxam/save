#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int f[21],dp[111];
int n;
int main(){
    memset(dp,127,sizeof(dp));
    for(int i=1;i<=10;i++)
        scanf("%d",&f[i]);
    scanf("%d",&n);
    dp[0]=0;
    dp[1]=f[1];
    for (int i=2;i<=n;i++){
        dp[i]=dp[i-1]+f[1];
        for (int j=2;j<=10 && j<=i;j++)
        dp[i]=min(dp[i],dp[i-j]+f[j]);
    }
    printf("%d",dp[n]);
    return 0;
}
