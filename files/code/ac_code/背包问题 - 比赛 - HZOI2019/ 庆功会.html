#include<cstdio>
#include<iostream>
using namespace std;
int m,n;
int w[505],c[505],p[505];
int f[6005];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d%d%d",&w[i],&c[i],&p[i]);
    for(int i=1;i<=n;i++){
        for(int j=1;j<=p[i];j++)
            for(int k=m;k>=w[i];k--)
                f[k]=max(f[k],f[k-w[i]]+c[i]);
    }
    printf("%d",f[m]);
    return 0;
}
