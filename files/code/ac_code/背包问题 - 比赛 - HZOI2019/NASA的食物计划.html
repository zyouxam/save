#include <bits/stdc++.h>  
using namespace std;  
int dp[450][450];
struct haha{
	int v;
	int m;
	int ka;
}food[51];
int main(){
	int vmax,mmax,n;
	scanf("%d %d %d",&vmax,&mmax,&n);
	for(int i=0;i<n;i++)
		scanf("%d %d %d",&food[i].v,&food[i].m,&food[i].ka);
	for(int i=0;i<n;i++){
		for(int j=vmax;j>=food[i].v;j--){
			for(int k=mmax;k>=food[i].m;k--)
				dp[j][k]=max(dp[j][k],dp[j-food[i].v][k-food[i].m]+food[i].ka);
		}
	}
	printf("%d\n",dp[vmax][mmax]);
	return 0;
}
