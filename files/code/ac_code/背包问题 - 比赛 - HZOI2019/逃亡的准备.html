#include<cstdio>
#include<iostream>
using namespace std;
int m,n;
int w[10001],c[10001],p[10001];
int f[10001];
void Bag(int n,int V,int i){
    int tot=0;
    for(int k=1;(tot+k<=p[i]) && (k*w[i]<=V);k<<=1){
        tot+=k;
        for(int j=V;j>=k*w[i];j--)
            f[j]=max(f[j],f[j-k*w[i]]+k*c[i]);
    }
    int x=p[i]-tot;
    if(x){
        for(int j=V;j>=x*w[i];j--)
            f[j]=max(f[j],f[j-x*w[i]]+x*c[i]);
    }
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d%d%d",&w[i],&c[i],&p[i]);
 
    for(int i=1;i<=n;i++){
        if(p[i]==0){
            for(int j=w[i];j<=m;j++){
                f[j]=max(f[j],f[j-w[i]]+c[i]);
            }
        }
        else{
            Bag(n,m,i);
        }
        //for(int j=1;j<=m;j++)
        //    printf("%5d ",f[j]);
        //printf("[%d]\n",i);
    }
    printf("%d",f[m]);
    return 0;
}
