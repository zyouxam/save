#include<cstdio>
#define MAXN int(1e6+10)
int dp[MAXN];
int w[MAXN];
int a[] = { 0,1,2,3,5,10,20 };
int main()
{
    int n = 0, k;
    for (int i = 1; i <= 6;++i)
    {
        scanf("%d", &k);
        for (int j = 1; j <= k; ++j)
            w[++n] = a[i];
    }
    int v = 0;
    for (int i = 1; i <= n; ++i)
        v += w[i];
    dp[0] = 1;
    for (int i = 1; i <= n;++i)
    for (int j = v; j >= w[i]; --j)
        dp[j] += dp[j - w[i]];
    int ans = 0;
    for (int i = 1; i <= v;++i)
    if (dp[i])++ans;
    printf("Total=%d\n", ans);
    return 0;
}
