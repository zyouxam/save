#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int m,n;
const int maxn=1e5+7;
long long dp[maxn];
int vis[25][maxn];
int w[25];
int main(){
	freopen("cd.in","r",stdin);
	freopen("cd.out","w",stdout);
    while(cin>>m>>n){
        memset(dp,0,sizeof(dp));
        memset(vis,0,sizeof(vis));
        for(int i=0;i<n;i++) cin>>w[i];
        for(int i=0;i<n;i++){
            for(int j=m;j>=w[i];j--){
                if(dp[j-w[i]]+w[i]>=dp[j]){
                    dp[j]=dp[j-w[i]]+w[i];
                    vis[i][j]=1;
                } 
            } 
        } 
        int w1[25],q=0;
        for(int i=n-1,j=m;i>=0;i--){
            if(vis[i][j]){
                w1[q++]=w[i];
                j=j-w[i];
            }
        }
        for(int i=q-1;i>=0;i--){
            cout<<w1[i]<<" ";
        } 
        printf("sum:%lld\n",dp[m]); 
    } 
    return 0; 
}
