#include <cstdio>
#include <iostream>
using namespace std;
const int maxn=30,maxc=30000+5;
int c[maxn],im[maxn],f[maxn][maxc];
int main(){
    int m,n;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++)
        scanf("%d%d",&c[i],&im[i]);
    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++){
           if(j<c[i])f[i][j]=f[i-1][j];
           else f[i][j]=max(f[i-1][j],f[i-1][j-c[i]]+c[i]*im[i]);
        }
    printf("%d\n",f[m][n]);
    return 0;
}
