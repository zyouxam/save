#include<cstdio>
#include<algorithm>
using namespace std;
int v[31],c[31],f[201];
int V,n;
int main(){
    scanf("%d%d",&V,&n);
    for(int i=1; i<=n; i++)scanf("%d%d",&v[i],&c[i]);
    for(int i=1; i<=n; i++)
        for(int j=v[i];j<=V; j++)
            f[j]=max(f[j],f[j-v[i]]+c[i]);
    printf("max=%d\n",f[V]);
    return 0;
}
