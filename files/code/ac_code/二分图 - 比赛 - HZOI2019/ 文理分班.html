#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>
using namespace std;
const int maxn = 1e3 + 500, INF = 0x3f3f3f3f;
int n, m, t, vis[maxn], match[maxn], a[maxn], b[maxn], p[maxn][maxn];
vector<int> g[maxn];
bool find(int u) {
    for (int i = 0; i < g[u].size(); i++) {
        int v = g[u][i];
        if (!vis[v]) {
            vis[v] = 1;
            if (match[v] == 0 || find(match[v])) {
                match[v] = u;
                return 1;
            }
        }
    }
    return 0;
}
int main() {
    scanf("%d", &t);
    while (t--) {
        for (int i = 0; i < maxn; i++) g[i].clear();
        scanf("%d", &n);
        for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
        for (int i = 1; i <= n; i++) {
            scanf("%d", &b[i]);
            if (b[i] != 0 && b[i] != 1) b[i] = 0;
            if (a[i] == 1 && b[i] == 0) g[i].push_back(i);
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                int x;
                scanf("%d", &x);
                if (x == 1 && a[j] == 1) {
                    g[i].push_back(j);
                }
            }
        }
        bool flag = 0;
        memset(match, 0, sizeof(match));
        for (int i = 1; i <= n; i++) {
            memset(vis, 0, sizeof(vis));
            if (!a[i] || (a[i] && !b[i])) {
                if (!find(i)) {
                    cout << "T_T" << endl;
                    flag = 1;
                    break;
                }
            }
        }
        if (flag == 0) cout << "^_^" << endl;
    }
}
