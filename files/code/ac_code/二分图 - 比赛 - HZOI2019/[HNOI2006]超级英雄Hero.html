#include <cstdio>
#include <cstring>
const int N = 1005;
bool map[N][N], vis[N];
int l[N], n, m, ans, sum[N];
bool Find(int x) {
    for (int j = 1; j <= n; j++)
        if (map[x][j] && !vis[j]) {
            vis[j] = true;
            if (l[j] == 0 || Find(l[j])) return l[j] = x, sum[x] = j, 1;
        }
    return false;
}
int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1, x, y; i <= m; i++) {
        scanf("%d%d", &x, &y);
        map[i][x + 1] = map[i][y + 1] = true;
    }
    for (int i = 1; i <= m; i++) {
        memset(vis, 0, sizeof(vis));
        if (Find(i)) ans++;
        else break;
    }
    printf("%d\n", ans);
    /// for (int i = 1; i <= ans; i++) printf("%d\n", sum[i] - 1);
}
