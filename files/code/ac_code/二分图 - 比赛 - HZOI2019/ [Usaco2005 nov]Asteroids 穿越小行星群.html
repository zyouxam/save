#include <algorithm>
#include <cstdio>
#include <cstring>
using namespace std;
const double eps = 1e-9;
int head[505], to[10005], nxt[10005], tot, n, k, link[505], ans;;
inline void add(int x, int y) {
    to[++tot] = y, nxt[tot] = head[x], head[x] = tot;
}
bool cover[505];
bool find(int k) {
    for (int i = head[k]; i; i = nxt[i]) {
        if (!cover[to[i]]) {
            int q = link[to[i]];
            link[to[i]] = k;
            cover[to[i]] = 1;
            if (q == -1) {
                ans++;
                return true;
            }
            if (find(q)) return true;
            link[to[i]] = q;
        }
    }
    return false;
}
int main() {
    scanf("%d%d", &n, &k);
    for (int i = 0, x, y; i < k; i++) {
        scanf("%d%d", &x, &y);
        add(x, y);
    }
    memset(link, -1, sizeof link);
    for (int i = 1; i <= n; i++) {
        memset(cover, 0, sizeof cover);
        find(i);
    }
    printf("%d\n", ans);
    return 0;
}
