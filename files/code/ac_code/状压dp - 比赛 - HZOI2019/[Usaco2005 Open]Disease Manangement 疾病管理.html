#include <iostream>
#include <cstdio>
#include <cstring>
#include <bitset>
using namespace std;
#define gc getchar()
inline int read(){
    int res=0;char ch=gc;
    while(!isdigit(ch))ch=gc;
    while(isdigit(ch)){res=(res<<3)+(res<<1)+(ch^48);ch=gc;}
    return res;
}
int n, m, k;
int eve[1005];
int f[1<<16];
int ans;

int main()
{
    n = read(), m = read(), k = read();
    for (int i = 1 ; i <= n ; i ++)
    {
        int x = read();
        while(x--)
        {
            int y = read();
            eve[i] |= (1<<(y-1));
        }
    }
    for (int i = (1<<m)-1 ; i >= 0 ; i --)
    {
        for (int j = 1 ; j <= n ; j ++)
        {
            f[i|eve[j]] = max(f[i|eve[j]], f[i] + 1);
        }
    }
    for (int i = 0 ; i <= (1<<m)-1 ; i ++)
    {
        bitset <16> bit = i;
        if (bit.count() > k) continue; 
        ans = max(ans, f[i]);
    }
    cout << ans << endl;
    return 0;
}
