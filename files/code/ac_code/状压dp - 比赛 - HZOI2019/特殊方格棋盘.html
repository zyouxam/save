#include <cstdio>
using namespace std;
long long f[1 << 20], s[25];
int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 0, a, b; i < m; i++) {
        scanf("%d%d",&a,&b);
        s[a] += 1 << (b - 1);
    }
    f[0] = 1;
    for (long long i = 1; i <= (1 << n); i++) {
        long long num = 0;
        for (long long t = i; t > 0; t -= t & -t) num++;
        for (long long t = i; t > 0; t -= t & -t)
            if (!(s[num] & (t & -t))) f[i] += f[i & ~(t & -t)];
    }
    printf("%lld\n",f[(1 << n) - 1]);
    return 0;
}
