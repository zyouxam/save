#include <bits/stdc++.h>
using namespace std;
int n, m, F[105], f[105][66][66], start[70], cnt = 0, gs[200];  
bool mp[105][30];
int main() {
    cin >> n >> m;
    char a;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> a;
            if (a == 'H') mp[i][j] = 1;  
        }
    }
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            F[i] = (F[i] << 1) + mp[i][j];
    start[++cnt] = 0;
    for (int i = 1; i < (1 << m); i++) {
        if (i & (i << 1) || i & (i << 2) || i & (i >> 1) ||i & (i >> 2)) continue;  
        start[++cnt] = i;  
        int x = i;
        while (x) {  
            gs[cnt]++;
            x -= (x & (-x));
        }
    }
    for (int i = 1; i <= cnt; i++)
        if (!(start[i] & F[1])) 
            f[1][i][0] = gs[i];
    for (int i = 1; i <= cnt; i++)
        if (!(start[i] & F[2]))
            for (int j = 1; j <= cnt; j++)
                if (!(start[i] & start[j]) && !(start[j] & F[1]))
                    f[2][i][j] = gs[j] + gs[i];
    for (int i = 3; i <= n; i++)
        for (int j = 1; j <= cnt; j++)
            if (!(start[j] & F[i]))
                for (int k1 = 1; k1 <= cnt; k1++)
                    if (!(start[j] & start[k1]) && !(start[k1] & F[i - 1]))
                        for (int k2 = 1; k2 <= cnt; k2++)
                            if (!(start[j] & start[k2])&& !(start[k1] & start[k2]) && !(start[k2] & F[i - 2]))
                                f[i][j][k1] = max(f[i][j][k1],f[i - 1][k1][k2] + gs[j]);
    int ans = 0;
    for (int i = 1; i <= cnt; i++)
        for (int j = 1; j <= cnt; j++)
            ans = max(ans, f[n][i][j]);
    cout << ans;
}
