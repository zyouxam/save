#include<iostream>
#include<cstring>
#include<cmath>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,c,ans;
int love[100005][38],f[100005][38];
int MAX(int a,int b){return a>b?a:b;}
int main()
{
    scanf("%d%d",&n,&c);
    for(int i=1;i<=c;i++)
    {
        int e,f,l;
        scanf("%d%d%d",&e,&f,&l);
        int like=0,hate=0;
        for(int j=1;j<=f;j++)
        {
            int x;
            scanf("%d",&x);
            x=(x-e+n)%n;
            like|=(1<<x);
        }
        for(int j=1;j<=l;j++)
        {
            int y;
            scanf("%d",&y);
            y=(y-e+n)%n;
            hate|=(1<<y);
        }
        for(int j=0;j<=31;j++)
            if((hate&j || ((31^j)&like)))
                love[e][j]++;
    }
    ans=0;
    for(int h=0;h<16;h++)
    {
        memset(f[0],128,sizeof(f[0]));
        f[0][h<<1]=0;
        for(int i=1;i<=n;i++)
        {
            for(int j=0;j<=31;j++)
            {
                f[i][j]=MAX(f[i-1][(j&15)<<1],f[i-1][(j&15)<<1|1])+love[i][j];                
            }
        }            
        ans=MAX(ans,MAX(f[n][h<<1],f[n][h<<1|1]));
    }  
    printf("%d",ans);
}
