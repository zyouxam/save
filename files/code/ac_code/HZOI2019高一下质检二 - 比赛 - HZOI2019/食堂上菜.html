#include <bits/stdc++.h>
using namespace std;
const int N = 20000 * 5 + 10;
int a[N], b[N], f[N], c[N], n, ans;
vector<int> p[N / 5];
void add(int x, int y) {
    for (; x <= n; x += (x & -x)) c[x] = max(c[x], y);
}
int ask(int x) {
    int aans = 0;
    for (; x; x -= (x & -x)) aans = max(aans, c[x]);
    return aans;
}
inline int input() {
    int t;
    scanf("%d", &t);
    return t;
}
int main() {
    freopen("dinning.in", "r", stdin);
    freopen("dinning.out", "w", stdout);
    n = 5 * input();
    for (int i = 1; i <= n; i++) p[a[i] = input()].push_back(i);
    for (int i = 1; i <= n; i++) b[i] = input();
    for (int i = 1; i <= n; i++) {
        int x = b[i];
        for (int j = p[b[i]].size() - 1; j >= 0; j--) {
            int pos = p[b[i]][j];
            f[pos] = ask(pos - 1) + 1;
            add(pos, f[pos]);
        }
    }
    for (int i = 1; i <= n; i++) ans = max(ans, f[i]);
    printf("%d\n", ans);
    return 0;
}
