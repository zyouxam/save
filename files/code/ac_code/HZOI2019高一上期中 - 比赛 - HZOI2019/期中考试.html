#include<cstdio>
#include<algorithm>
using namespace std;
long long s[1000005];
int main(){
        freopen("exam.in","r",stdin);
        freopen("exam.out","w",stdout);
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++){
                long long a,b,c;
                scanf("%lld%lld%lld",&a,&b,&c);
                s[i]=a+b+c;
        }
        sort(s,s+n);
        
        
        int q;
        scanf("%d",&q);
        for(int i=0;i<q;i++){
                long long key;
                scanf("%lld",&key);
                if(key>=s[n-1]){
                        printf("0\n");
                        continue;
                }
                else if(key<s[0]){
                        printf("%d\n",n);
                        continue;
                }
                int l=0,r=n-1,m;
                while(l<r-1){
                        m=(l+r)/2;
                        if(s[m]>key){
                                r=m;
                        }
                        else{
                                l=m;
                        }
                }
                printf("%d\n",n-(l+1));
        }
        return 0;
}
