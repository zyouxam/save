#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int d[6][6], cnt, c[6][6], ans=0x7f7f7f7f;
bool cal(int a){
	cnt=0;
	for(int i=2;i<=3;i++)
		for(int j=2;j<=3;j++)
			if(c[i-1][j]!=a&&c[i][j]!=a&&c[i][j-1]!=a&&c[i][j+1]!=a&&c[i+1][j]!=a)
				c[i-1][j]^=1, c[i][j]^=1, c[i][j-1]^=1, c[i][j+1]^=1, c[i+1][j]^=1, cnt++;
	for(int i=1;i<=4;i++)
		for(int j=1;j<=4;j++)
			if(c[i-1][j]!=a&&c[i][j]!=a&&c[i][j-1]!=a&&c[i][j+1]!=a&&c[i+1][j]!=a)
				c[i-1][j]^=1, c[i][j]^=1, c[i][j-1]^=1, c[i][j+1]^=1, c[i+1][j]^=1, cnt++;
	for(int i=2;i<=4;i++)
		for(int j=1;j<=4;j++)
			if(c[i-1][j]!=a) c[i-1][j]^=1, c[i][j]^=1, c[i][j-1]^=1, c[i][j+1]^=1, c[i+1][j]^=1, cnt++;
	for(int i=1;i<=4;i++)
		for(int j=1;j<=4;j++)
			if(c[i][j]!=a) return 0;
	return 1;
}
int main(){
	char s[6];
	for(int i=1,j;i<=4;i++) for(scanf("%s",s), j=0;j<4;j++) d[i][j+1]=(s[j]=='b')?1:0;
	for(int i=1;i<=4;i++) for(int j=1;j<=4;j++) c[i][j]=d[i][j];
	ans=cal(1)?min(ans,cnt):ans, memset(c,1,sizeof(c));
	for(int i=1;i<=4;i++) for(int j=1;j<=4;j++) c[i][j]=d[i][j];
	ans=cal(0)?min(ans,cnt):ans, memset(c,0,sizeof(c));
	for(int i=1;i<=4;i++) for(int j=1;j<=4;j++) c[i][j]=d[5-i][j];
	ans=cal(1)?min(ans,cnt):ans, memset(c,1,sizeof(c));
	for(int i=1;i<=4;i++) for(int j=1;j<=4;j++) c[i][j]=d[5-i][j];
	ans=cal(0)?min(ans,cnt):ans;
	if(ans==0x7f7f7f7f) printf("Impossible\n");
	else printf("%d\n",ans);
	return 0;
}
