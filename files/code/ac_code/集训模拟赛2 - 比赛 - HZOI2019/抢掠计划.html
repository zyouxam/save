#include <algorithm>
#include <cstdio>
#include <cstring>
#include <queue>
using namespace std;
const int N = 1e7 + 10, INF = 0x3f3f3f3f;
struct Edge { int to, next; } e[N], E[N];
int len, Head[N], tto[N], fro[N], p[N], t[N];
void Ins(int a, int b) {
    e[++len].to = b;
    e[len].next = Head[a];
    Head[a] = len;
}
int l, H[N];
void Ins2(int a, int b) {
    E[++l].to = b;
    E[l].next = H[a];
    H[a] = l;
}
int low[N], dfn[N], stk[N], top, num, belong[N], cnt;
long long w[N];
void tarjan(int u) {
    low[u] = dfn[u] = ++num;
    stk[++top] = u;
    for (int x = Head[u]; x; x = e[x].next) {
        int v = e[x].to;
        if (!dfn[v]) {
            tarjan(v);
            low[u] = min(low[u], low[v]);
        } else if (!belong[v])
            low[u] = min(low[u], dfn[v]);
    }
    if (low[u] == dfn[u]) {
        ++cnt;
        while (1) {
            int x = stk[top--];
            belong[x] = cnt;
            w[cnt] += p[x];
            if (x == u) break;
        }
    }
}
bool vis[N];
long long dis[N];
void spfa(int s, int n) {
    queue<int> q;
    for (int i = 1; i <= n; i++) dis[i] = -INF, vis[i] = 0;
    dis[s] = w[s];
    q.push(s);
    vis[s] = 1;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        vis[u] = 0;
        for (int i = H[u]; i; i = E[i].next) {
            int v = E[i].to;
            if (dis[v] < dis[u] + w[v]) {
                dis[v] = dis[u] + w[v];
                if (!vis[v]) {
                    q.push(v);
                    vis[v] = 1;
                }
            }
        }
    }
}
int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; i++) {
        scanf("%d%d", &fro[i], &tto[i]);
        Ins(fro[i], tto[i]);
    }
    for (int i = 1; i <= n; i++) {
        scanf("%d", &p[i]);
    }
    int s, P;
    for (int i = 1; i <= n; i++)
        if (!dfn[i]) tarjan(i);
    scanf("%d%d", &s, &P);
    s = belong[s];
    int pnum = 0;
    for (int i = 1; i <= P; i++) {
        int a;
        scanf("%d", &a);
        if (!vis[belong[a]]) {
            t[++pnum] = belong[a];
            vis[belong[a]] = 1;
        }
    }
    for (int i = 1; i <= m; i++) {
        int u = belong[fro[i]], v = belong[tto[i]];
        if (u != v) Ins2(u, v);
    }
    spfa(s, cnt);
    long long ans = -1;
    for (int i = 1; i <= pnum; i++) {
        ans = max(ans, dis[t[i]]);
    }
    printf("%lld\n", ans);
}
