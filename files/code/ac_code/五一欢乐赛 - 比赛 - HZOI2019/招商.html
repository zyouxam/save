#include <cstdio>
#include <stack>
using namespace std;
const int maxn = 400000 + 5;
int h[maxn], n;
int main() {
    freopen("post.in","r",stdin);
    freopen("post.out","w",stdout);
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) scanf("%d", &h[i]);
    stack<int> s;
    long long ans = 0;
    s.push(0);
    h[0] = h[++n] = 0;
    for (int i = 1; i <= n; ++i) {
        while (h[i] < h[s.top()]) {
            long long height = h[s.top()];
            s.pop();
            long long width = i - s.top() - 1;
            if (width * height > ans) ans = width * height;
        }
        s.push(i);
    }
    printf("%d\n", ans);
    return 0;
}
