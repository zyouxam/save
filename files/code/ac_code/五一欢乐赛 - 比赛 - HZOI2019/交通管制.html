#include <cstdio>
#include <cstring>
#include <iostream>
#define M 500005
#include <queue>
#define N 255
using namespace std;
int n, m, last, ans, head[N], to[M], nxt[M], len[M], e = 1, dis[N], vis[N], pree[N], nexte[N], that[M], nu;
void add(int u, int v, int l) { nxt[++e] = head[u], head[u] = e, to[e] = v, len[e] = l; }
queue<int> q;
void spfa(int s) {
    memset(dis, 20, sizeof(dis));
    dis[s] = 0, vis[s] = 1, q.push(s);
    while (!q.empty()) {
        int now = q.front();
        q.pop();
        vis[now] = 0;
        for (int i = head[now]; i; i = nxt[i]) {
            int j = to[i];
            if (dis[j] > dis[now] + len[i]) {
                dis[j] = dis[now] + len[i];
                pree[j] = i;
                nexte[j] = now;
                if (!vis[j]) vis[j] = 1, q.push(j);
            }
        }
    }
}
int main() {
    freopen("traffic.in","r",stdin);
    freopen("traffic.out","w",stdout);
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; ++i) {
        int u, v, l;
        scanf("%d%d%d", &u, &v, &l);
        add(u, v, l), add(v, u, l);
    }
    spfa(1);
    last = dis[n];
    int now = n;
    while (now != 1) that[++nu] = pree[now], now = nexte[now];
    for (int i = 1; i <= nu; ++i) {
        len[that[i]] *= 2, len[that[i] ^ 1] *= 2;
        spfa(1), ans = max(ans, dis[n]);
        len[that[i]] /= 2, len[that[i] ^ 1] /= 2;
    }
    printf("%d\n", ans - last);
    return 0;
}
