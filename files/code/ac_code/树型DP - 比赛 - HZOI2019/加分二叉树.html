#include <cstdio>
int f[33][33], r[33][33], a[33], n;
void front(int p, int q) {
    if (p > q || p < 1 || q > n) return;
    printf("%d ", r[p][q]);
    front(p, r[p][q] - 1);
    front(r[p][q] + 1, q);
}
int main() {
    int i, j, k, t, s;
    scanf("%d", &n);
    for (i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        f[i][i] = a[i];
        r[i][i] = i;
        f[i][i - 1] = 1;
    }
    f[n + 1][n] = 1;
    for (i = 1; i < n; i++)
        for (j = 1; j <= n - i; j++)
            for (k = j; k <= j + i; k++) {
                t = a[k] + f[j][k - 1] * f[k + 1][j + i];
                if (t > f[j][j + i]) {
                    f[j][j + i] = t;
                    r[j][j + i] = k;
                }
            }
    printf("%d\n", f[1][n]);
    front(1, n);
    return 0;
}
