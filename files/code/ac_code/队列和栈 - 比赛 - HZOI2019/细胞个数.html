#include<cstdio>
#include<queue>
using namespace std;
const int N=70;
int a[N][N];
int d[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
int m,n;
struct P{
        int x,y;
        P(int a,int b){
                x=a;
                y=b;
        }
};
int bfs(int x,int y){
        queue<P> q;
        q.push(P(x,y));
        a[x][y]=0;
        while(!q.empty()){
                P p=q.front();
                q.pop();
                for(int i=0;i<4;i++){
                        int nx=p.x+d[i][0],ny=p.y+d[i][1];
                        if(nx<0 || nx>=m || ny<0 || ny>=n || !a[nx][ny])continue;
                        a[nx][ny]=0;
                        q.push(P(nx,ny));
                        //printf("%d %d\n",nx,ny);
                }
        }
}
int solve(){
        int ans=0;
        for(int i=0;i<m;i++){
                for(int j=0;j<n;j++){
                        if(a[i][j]){
                                bfs(i,j);
                                ans++;
                        }
                }
        }
        return ans;
}
int main(){
		freopen("cell.in","r",stdin);
		freopen("cell.out","w",stdout);
        scanf("%d%d",&m,&n);
        for(int i=0;i<m;i++){
                for(int j=0;j<n;j++){
                        scanf("%1d",&a[i][j]);
                }
        }
        printf("%d\n",solve());
        return 0;
}
