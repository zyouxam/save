#include<cstdio>
#include<stack>
using namespace std;
int judge(char a,int ifInStack){
        if(a=='+' || a=='-')return 1;
        if(a=='*' || a=='/')return 2;
        if(a=='(' && ifInStack)return 0;
        if(a=='(' && !ifInStack)return 3;
        if(a==')')return 4;  
        return -1;
}
int main(){
	freopen("suffix.in","r",stdin);
	freopen("suffix.out","w",stdout);
	char s[260];
	stack<char> stk;
	scanf("%s",s);
	for(int i=0;i<260;i++){
	        if(!s[i])break;
	        int pro=judge(s[i],0);
	        if(pro<0){
	                printf("%c",s[i]);
	                continue;
	        }
	        if(s[i]==')'){
	                while(!stk.empty() && stk.top()!='('){
	                        printf("%c",stk.top());
	                        stk.pop();
	                }
	                if(!stk.empty())stk.pop();
	        }     
	        else{
	                while(!stk.empty() && judge(stk.top(),1)>=pro){
	                        printf("%c",stk.top());
	                        stk.pop();
	                }
	                stk.push(s[i]);
	        }  
	}
	while(!stk.empty()){
                printf("%c",stk.top());
                stk.pop();
        }
	return 0;
}
