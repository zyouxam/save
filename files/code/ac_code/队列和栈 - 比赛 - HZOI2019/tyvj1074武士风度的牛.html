#include<cstdio>
#include<cstring>
#include<queue>
using namespace std;
struct node{
        int x;
        int y;
        int step;
};
int main(){
	freopen("bull.in","r",stdin);
	freopen("bull.out","w",stdout);
	int m,n;
	int d[8][2]={{-1,-2},{-2,-1},{-2,1},{-1,2},{1,2},{2,1},{2,-1},{1,-2}};
	scanf("%d %d",&n,&m);
	char line[155]={0};
	int map[155][155]={0};
	int bx,by,ex,ey;
	for(int i=0;i<m;i++){
	        scanf("%s",line);
	        for(int j=0;j<n;j++){
	                if(line[j]=='*')map[i][j]=1;
	                else if(line[j]=='K'){
	                        bx=i;
	                        by=j;
	                }
	                else if(line[j]=='H'){
	                        ex=i;
	                        ey=j;
	                }
	        }
	}
	node start;
	start.x=bx;
	start.y=by;
	queue<node> q;
	q.push(start);
	int visited[155][155]={0};
	while(!q.empty()){
	        node now=q.front();
	        q.pop();
	        for(int i=0;i<8;i++){
	                int nx=now.x+d[i][0];
	                int ny=now.y+d[i][1];
	                int step=now.step+1;
	                if(nx==ex && ny==ey){
	                        printf("%d\n",step);
	                        return 0;
	                }
	                node next;
	                next.x=nx;
	                next.y=ny;
	                next.step=step;
	                if(!visited[nx][ny] && nx>=0 && nx<m && ny>=0 && ny<=n){
        	                visited[nx][ny]=1;
        	                q.push(next);
	                }
	        }
	}
	return 0;
}
