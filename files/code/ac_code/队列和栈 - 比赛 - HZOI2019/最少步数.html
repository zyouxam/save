#include<cstdio>
#include<queue>
using namespace std;
struct node{
        int x;
        int y;
        int step;
        node(int a,int b,int s){
                x=a;
                y=b;
                step=s;
        }
};
int main(){
	freopen("step.in","r",stdin);
	freopen("step.out","w",stdout);
	int m=100,n=100;
	int info[2][2];
	int d[12][2]={{-1,-2},{-2,-1},{-2,1},{-1,2},{1,2},{2,1},{2,-1},{1,-2},{2,2},{2,-2},{-2,2},{-2,-2}};
	int map[155][155]={0};
	scanf("%d %d %d %d",&info[0][0],&info[0][1],&info[1][0],&info[1][1]);
	
	for(int i=0;i<2;i++){
	        queue<node> q;
	        q.push(node(info[i][0],info[i][1],0));
	        int visited[155][155]={0};
	        int flag=1;
	        while(!q.empty()){
	                node now=q.front();
	                q.pop();
	                for(int i=0;i<12;i++){
	                        int nx=now.x+d[i][0];
	                        int ny=now.y+d[i][1];
	                        int step=now.step+1;
	                        if(nx==1 && ny==1){
	                                printf("%d\n",step);
	                                flag=0;
	                                break;
	                        }
	                        if(!visited[nx][ny] && nx>=0 && nx<m && ny>=0 && ny<=n){
                	                visited[nx][ny]=1;
                	                q.push(node(nx,ny,step));
	                        }
	                }
	                if(!flag)break;
	        }
        }
	return 0;
}
