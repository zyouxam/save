#include<cstdio>
#include<stack>
using namespace std;
int main(){
	freopen("check.in","r",stdin);
	freopen("check.out","w",stdout);
	char s[300]={0};
	stack<char> stk;
	scanf("%s",s);
	for(int i=0;i<300;i++){
		if(!s[i])break;
		if(s[i]=='[')stk.push('[');
		else if(s[i]=='(')stk.push('(');
		else if(s[i]==']'){
			if(stk.empty() || stk.top()!='['){
				printf("Wrong");
				return 0;
			}
			else stk.pop();
		}
		else if(s[i]==')'){
			if(stk.empty() || stk.top()!='('){
				printf("Wrong");
				return 0;
			}
			else stk.pop();
		}
	}
	if(stk.empty())printf("OK");
	else printf("Wrong");
	return 0;
}
