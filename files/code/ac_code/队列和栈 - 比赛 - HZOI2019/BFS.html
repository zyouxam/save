#include<cstdio>
#include<queue>
using namespace std;
int main(){
	freopen("bfs.in","r",stdin);
	freopen("bfs.out","w",stdout);
	int n,e;
	int map[100][100]={};
	int visited[100]={0,1};
	queue<int> q;
	scanf("%d%d",&n,&e);
	for(int i=0;i<e;i++){
	        int a,b;
	        scanf("%d%d",&a,&b);
	        map[a][b]=map[b][a]=1;
	}
	q.push(1);
	printf("1 ");
	while(!q.empty()){
	        int now=q.front();
	        q.pop();
	        for(int i=1;i<=n;i++){
	                if(map[now][i] && !visited[i]){
                                visited[i]=1;
                                q.push(i);
                                printf("%d ",i);
	                }
	        }
	}
	return 0;
}
