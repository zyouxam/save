#include<cstdio>
#include<stack>
using namespace std;
char b[8]={'(',')','[',']','{','}','<','>'};
int what(char a){
	int flag=false;
	for(int i=0;i<8;i++)if(b[i]==a)flag=true;
	if(!flag)return 0;
	else if(a=='(' || a=='[' || a=='{' || a=='<')return 1;
	return 2;
}
int match(char a,char b){
	if((a=='[' && b==']')||(a=='<' && b=='>')||(a=='{' && b=='}')||(a=='(' && b==')'))return 1;
	return 0;
}
int main(){
	freopen("strc.in","r",stdin);
	freopen("strc.out","w",stdout);
	int n;
	int data[128]={0};
	data['<']=1;data['(']=2;data['[']=3;data['{']=4;
	scanf("%d",&n);
	for(int l=0;l<n;l++){
		char s[300];
		stack<char> stk;
		scanf("%s",s);
		bool ifright=true;
		for(int i=0;i<300;i++){
			if(!s[i])break;
			if(!what(s[i]))continue;
			if(stk.empty() && what(s[i])==1){
				stk.push(s[i]);	
			}
			else if(what(s[i])==1){
				if(what(stk.top())==1 && data[stk.top()]<data[s[i]]){
					ifright=false;
					//printf("E:1 %d:%c %c\n",i,s[i],stk.top());
					break;
				}
				stk.push(s[i]);
			}
			else if(what(s[i])==2){
				if(stk.empty() || !match(stk.top(),s[i])){
					//printf("E:2 %d:%c\n",i,s[i]);
					ifright=false;
					break;
				}
				stk.pop();
			}
		}
		if(!stk.empty())ifright=false;
		if(ifright)printf("YES\n");
		else printf("NO\n");
		
	}
	return 0;
}
