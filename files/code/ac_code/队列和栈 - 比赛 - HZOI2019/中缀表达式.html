using namespace std;
bool panduan(char s[]){
    for (int i=0;i<strlen(s);i++){
        if (s[i]>='0'&&s[i]<='9'){
            if ((s[i-1]>='0'&&s[i-1]<='9')||(s[i+1]>='0'&&s[i+1]<='9'))
                return false;
        }
        if (s[i]=='*'||s[i]=='/'||s[i]=='+'||s[i]=='-'){
            if (s[i-1]=='*'||s[i-1]=='/'||s[i-1]=='+'||s[i-1]=='-'||s[i-1]=='(')
                return false;
            if (s[i+1]=='*'||s[i+1]=='/'||s[i+1]=='+'||s[i+1]=='-'||s[i+1]==')')
                return false;
        }
        if (s[i]=='/'&&s[i+1]=='0')
            return false;
    }
    return true;
}
void zh(char s[],char r[]){
    int len=strlen(s);
    int jishu=0;
    stack<char> stk;
    for (int i=0;i<len;i++){
        if (s[i]>='0'&&s[i]<='9'){
            r[jishu]=s[i];
            jishu++;
        }
        else{
            if (stk.empty())
                stk.push(s[i]);
            else{
                if (s[i]=='+'||s[i]=='-'){
                    if (stk.top()=='(')
                        stk.push(s[i]);
                    else {
                        if (stk.top()=='*'||stk.top()=='/'){
                            while (!stk.empty()&&stk.top()!='('){
                                r[jishu]=stk.top();
                                jishu++;
                                stk.pop();
                            }
                            stk.push(s[i]);
                        }
                        else{
                            r[jishu]=stk.top();
                            jishu++;
                            stk.pop();
                            stk.push(s[i]);
                        }
                    }
                }
                if (s[i]=='*'||s[i]=='/'){
                    if (stk.top()=='(')
                        stk.push(s[i]);
                    else{
                        if (stk.top()=='+'||stk.top()=='-')
                            stk.push(s[i]);
                        else{
                            r[jishu]=stk.top();
                            jishu++;
                            stk.pop();
                            stk.push(s[i]);
                        }
                    }
                }
                if (s[i]=='(')
                    stk.push(s[i]);
                if (s[i]==')'){
                    while (!stk.empty()&&stk.top()!='('){
                        r[jishu]=stk.top();
                        jishu++;
                        stk.pop();
                    }
                    stk.pop();
                }
            }
        }
    }
    while (!stk.empty()){
        r[jishu]=stk.top();
        jishu++;
        stk.pop();
    }
}
int ys(char r[]){
    int len=strlen(r),a,b,c;
    stack<int> num;
    for (int i=0;i<len;i++){
        if (r[i]>='0'&&r[i]<='9')
            num.push(r[i]-'0');
        if (r[i]=='+'){
            a=num.top();
            num.pop();
            b=num.top();
            num.pop();
            c=a+b;
            num.push(c);
        }
        if (r[i]=='-'){
            a=num.top();
            num.pop();
            b=num.top();
            num.pop();
            c=b-a;
            num.push(c);
        }
        if (r[i]=='*'){
            a=num.top();
            num.pop();
            b=num.top();
            num.pop();
            c=a*b;
            num.push(c);
        }
        if (r[i]=='/'){
            a=num.top();
            num.pop();
            b=num.top();
            num.pop();
            c=a/b;
            num.push(c);
        }
    }
    return num.top();
}
char s[1000],r[1000];
int main(){
    freopen("infix.in","r",stdin);
    freopen("infix.out","w",stdout);
    scanf("%s",s);
    s[strlen(s)-1]='\0';
    int ans;
    if (!panduan(s))
        cout<<"NO";
    else{
        zh(s,r);
        ans=ys(r);
        /*printf("%s\n",r);*/
        cout<<ans;
    }
    return 0;
}
