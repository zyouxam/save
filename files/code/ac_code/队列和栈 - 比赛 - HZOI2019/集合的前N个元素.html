#include<cstdio>
#include<queue>
using namespace std;
inline void print(int n){printf("%d ",n);}
inline void print(char n){printf("%c",n);}
int main(){
        freopen("element.in","r",stdin);
        freopen("element.out","w",stdout);
        int n,c=1;
	queue<int> x,y;
	x.push(1);
	y.push(1);
        scanf("%d",&n);
	while(c++<=n){
	        int a=x.front(),b=y.front(),next;
                if(a<b){
                        print(a);
                        x.pop();
                        next=a;
                } 
                else if(a>b){
                        print(b);
                        y.pop();
                        next=b;
                }
                else{
                        print(a);
                        x.pop();y.pop();
                        next=a;
                }
                x.push(2*next+1);
                y.push(3*next+1);
	}
	return 0;
}
