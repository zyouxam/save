#include<cstdio>
int main(){
	freopen("bracket.in","r",stdin);
	freopen("bracket.out","w",stdout);
	char s[300]={0};
	scanf("%s",s);
	int top=0;
	for(int i=0;i<300;i++){
		if(!s[i] || s[i]=='@')break;
		if(s[i]=='(')top++;
		else if(s[i]==')')top--;
	}
	if(top==0)printf("YES");
	else printf("NO");
	return 0;
}
