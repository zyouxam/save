#include<cstdio>
#define MAXN 1000005
int main(){
    int n,m;
    int a,b;
    long long c;
    long long data[MAXN];
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%lld",&data[i]);
    }
    scanf("%d",&m);
    char o[10];
    for(int i=1;i<=m;i++){
        scanf("%s",o);
        if(o[0]=='A'){
            scanf("%d%d%lld",&a,&b,&c);
            for(int j=a;j<=b;j++){
                data[j]+=c;
            }
        }else{
            scanf("%d%d",&a,&b);
            long long sum=0;
            for(int j=a;j<=b;j++){
                sum+=data[j];
            }
            printf("%lld\n",sum);
        }
    }
    return 0;
}
