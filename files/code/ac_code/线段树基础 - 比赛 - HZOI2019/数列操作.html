#include <cstdio>
#define MAXN 100005
using namespace std;
int data[MAXN];
int tree[MAXN*4];
int n,m,a,b;
void build(int root,int l,int r){
    if(l==r){
        tree[root]=data[l];
        return;
    }else if(l>r)return;
    int mid=(l+r)>>1;
    build(root<<1,l,mid);
    build(root<<1|1,mid+1,r);
    tree[root]=tree[root<<1]+tree[root<<1|1];
}
void add(int root,int l,int r,int x,int w){
    if(l>r)return;
    else if(l==r){
        tree[root]+=w;
        return;
    }
    int mid=(l+r)>>1;
    if(x<=mid)add(root<<1,l,mid,x,w);
    else add(root<<1|1,mid+1,r,x,w);
    tree[root]=tree[root<<1]+tree[root<<1|1];
}
int query(int root,int l,int r,int s,int t){
    if(l>r)return 0;
    else if(l==r)return tree[root];
    int mid=(l+r)>>1;
    if(s<=l && t>=r)return tree[root];
    else if(t<=mid)return query(root<<1,l,mid,s,t);
    else if(s>mid)return query(root<<1|1,mid+1,r,s,t);
    else return query(root<<1,l,mid,s,t)+query(root<<1|1,mid+1,r,s,t);
}
int main(){
    char o[10];
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&data[i]);
    build(1,1,n);
    scanf("%d",&m);
    for(int i=1;i<=m;i++){
        scanf("%s%d%d",o,&a,&b);
        if(o[0]=='A'){
            add(1,1,n,a,b);
        }
        else printf("%d\n",query(1,1,n,a,b));
    }
    return 0;
}
