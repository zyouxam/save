#include <cstdio>
#define MAXN 100005
using namespace std;
long long data[MAXN];
long long tree[MAXN*4];
long long lazy[MAXN*4];
int n,m,a,b;
long long c;
void build(int root,int l,int r){
    if(l==r){
        tree[root]=data[l];
        return;
    }else if(l>r)return;
    int mid=(l+r)>>1;
    build(root<<1,l,mid);
    build(root<<1|1,mid+1,r);
    tree[root]=tree[root<<1]+tree[root<<1|1];
}
void update(int root,int l,int r,long long w) {
    tree[root]+=(r-l+1)*w;
    lazy[root]+=w;
}
void down(int root,int l,int r){
    int mid=(l+r)>>1;
    update(root<<1,l,mid,lazy[root]);
    update(root<<1|1,mid+1,r,lazy[root]);
    lazy[root]=0;
}
void add(int root,int l,int r,int s,int t,long long w){
    //if(l>s)return;
    if(s<=l&& t>=r){
        update(root,l,r,w);
        return;
    }
    int mid=(r+l)>>1;
    down(root,l,r);
    if(s<=mid)add(root<<1,l,mid,s,t,w);
    if(t>mid)add(root<<1|1,mid+1,r,s,t,w);
    tree[root]=tree[root<<1]+tree[root<<1|1];
}
long long query(int root,int l,int r,int s,int t){
    if(l>r)return 0;
    else if(l==r)return tree[root];
    int mid=(l+r)>>1;
    down(root,l,r);
    if(s<=l && t>=r)return tree[root];
    else if(t<=mid)return query(root<<1,l,mid,s,t);
    else if(s>mid)return query(root<<1|1,mid+1,r,s,t);
    else return query(root<<1,l,mid,s,t)+query(root<<1|1,mid+1,r,s,t);
}
int main(){
    char o[10];
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%lld",&data[i]);
    build(1,1,n);
    scanf("%d",&m);
    for(int i=1;i<=m;i++){
        scanf("%s",o);
        if(o[0]=='A'){
            scanf("%d%d%lld",&a,&b,&c);
            add(1,1,n,a,b,c);
        }
        else{
            scanf("%d",&a);
            printf("%lld\n",query(1,1,n,a,a));
        }
    }
    return 0;
}
