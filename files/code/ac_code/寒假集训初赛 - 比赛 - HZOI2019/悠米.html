#include<cstdio>
#include<cstring>
#include<queue>
#define N 4000005
#define M 40000005
using namespace std;
int n,m,k,u,v,w;
struct Edge{
    int next,to,dis;
}edge[M<<1];
int cnt=1,head[N];
struct Node{
    int u,dis;
    bool operator < (const Node &a) const{
        return a.dis<dis;
    }
    Node(int a,int b){
        u=a;
        dis=b;
    }
};
void add(int from,int to,int dis){
    edge[++cnt].next=head[from];
    edge[cnt].to=to;
    edge[cnt].dis=dis;
    head[from]=cnt;
}
int dis[N];
bool vis[N];
void Dijkstra(int s){
    memset(vis,0,sizeof(vis));
    memset(dis,0x3f,sizeof(dis));
    priority_queue<Node> q;
    q.push(Node(s,0));
    dis[s]=0;
    while(!q.empty()){
        int u=q.top().u;q.pop();
        if(vis[u]) continue;
        vis[u]=1;
        for(int i=head[u];i;i=edge[i].next){
            int v=edge[i].to;
            if(dis[v]>dis[u]+edge[i].dis){
                dis[v]=dis[u]+edge[i].dis;
                if(!vis[v]) q.push((Node){v,dis[v]});
            }
        }
    }
}

int main(){
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1;i<=m;++i){
        scanf("%d%d%d",&u,&v,&w);
        for(int j=0;j<=k;++j){
            add(u+j*n,v+j*n,w);
            add(v+j*n,u+j*n,w);
        }
        for(int j=0;j<k;++j){
            add(u+j*n,v+(j+1)*n,w/2);
            add(v+j*n,u+(j+1)*n,w/2);
        }
    }
    for(int i=0;i<k;++i)add(n+i*n,n+(i+1)*n,0);
    Dijkstra(1);
    printf("%d\n",dis[(k+1)*n]);
    return 0;
}
