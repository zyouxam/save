#include <cstdio>
int main() {
    int n = 0, a[1005], f[1005], dp[1005], maxv = 0, maxi;
    while (~scanf("%d", &a[n])) n++;
    for (int i = 0; i < n; i++) {
        dp[i] = 1;
        f[i] = i;
        for (int j = 0; j < i; j++) {
            if (a[j] < a[i] && dp[j] + 1 > dp[i]) {
                dp[i] = dp[j] + 1;
                f[i] = j;
            }
        }
        if(dp[i]>maxv){
            maxv=dp[i];
            maxi=i;
        }
    }
    printf("max=%d\n", maxv);
    int top=1,stack[1005];
    for(;f[maxi]!=maxi;maxi=f[maxi])stack[top++]=a[maxi];stack[top]=a[maxi];
    while(top)printf("%d ",stack[top--]);
    return 0;
}
