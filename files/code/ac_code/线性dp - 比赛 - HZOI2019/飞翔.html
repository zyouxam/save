#include <cstdio>
#include<algorithm>
using namespace std;
struct Point {
    int x, y;
    bool operator<(Point other)const{
        if(x!=other.x)
            return x<other.x;
        return y<other.y;
    }
} p[1005];
int main() {
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 1; i <= k; i++) scanf("%d%d", &p[i].x, &p[i].y);
    int dp[1005],maxv=0;
    sort(p+1,p+k+1);
    for (int i = 1; i <= k; i++) {
        dp[i]=1;
        for (int j = 1; j < i; j++){
            if(p[j].x<p[i].x && p[j].y<p[i].y && dp[j] + 1 > dp[i])
                dp[i]=dp[j]+1;
        }
        maxv=max(maxv,dp[i]);
    }
    double d=141.42135623730951;
    printf("%d",int((n+m-2*maxv)*100+d*maxv+0.5));
    return 0;   
}
