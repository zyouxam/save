#include <algorithm>
#include <cstring>
#include <iostream>
using namespace std;
char a[3000];
int n, d, dp[3000], hi, ji;
int main() {
    cin >> n >> d;
    memset(dp, 0x3f, sizeof(dp));
    dp[0] = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        for (int j = i; j >= 1; j--) {
            if (a[j] == 'H') hi++;
            if (a[j] == 'J') ji++;
            if (hi == 0 || ji == 0 || abs(hi - ji) <= d) {
                dp[i] = min(dp[i], dp[j - 1] + 1);
            }
        }
        hi = 0;
        ji = 0;
    }
    cout << dp[n];
    return 0;
}
