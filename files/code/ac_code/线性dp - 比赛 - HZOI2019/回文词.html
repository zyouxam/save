#include <bits/stdc++.h>
using namespace std;
#define maxn 5010

char str1[maxn], str2[maxn];
int f[maxn][maxn];

int main() {
    int len;
    scanf("%d",&len);
    scanf("%s", str1 + 1);
    int cnt = 0;
    for (int i = len; i >= 1; i--) {
        cnt++;
        str2[cnt] = str1[i];
    }
    memset(f, 0, sizeof(f));
    for (int i = 1; i <= len; i++) {
        for (int j = 1; j <= len; j++) {
            if (str1[i] == str2[j])
                f[i][j] = f[i - 1][j - 1] + 1;
            else
                f[i][j] = max(f[i - 1][j], f[i][j - 1]);
        }
    }
    printf("%d", len - f[len][len]);
    return 0;
}
