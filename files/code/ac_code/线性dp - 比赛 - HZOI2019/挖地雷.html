#include<cstdio>
struct Edge{
    int from,to,next;
} edge[40005];
int head[205],cnt;
void add(int x,int y){
    edge[++cnt].next=head[y];
    edge[cnt].from=x;
    edge[cnt].to=y;
    head[y]=cnt;
}
int main(){
    int n,x,y;
    scanf("%d", &n);
    int l[205];
    for(int i=1; i <= n; i++)scanf("%d",&l[i]);
    while(1){
        scanf("%d%d",&x,&y);
        if(!x&&!y)break;
        add(x,y);
    }
    int dp[205],f[205],maxv = 0, maxi;
    for (int i = 1; i <= n; i++) {
        dp[i]=l[i];
        f[i] = i;
        for(int x=head[i];x;x=edge[x].next){
            int j=edge[x].from;
            if (dp[j] + l[i] > dp[i]) {
                dp[i] = dp[j] + l[i];
                f[i] = j;
            }
        }
        if(dp[i]>maxv){
            maxv=dp[i];
            maxi=i;
        }
    }
    int top=1,stack[1005];
    for(;f[maxi]!=maxi;maxi=f[maxi])stack[top++]=maxi;stack[top]=maxi;
    while(top)printf("%d%s",stack[top--],(top!=1?"-":"\n"));
    printf("%d\n", maxv);
}
