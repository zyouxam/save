#include<stdio.h>
int a[10005]={0},dp[10005]={0},dp2[10005]={0},j=2,n=1;
int main(){
    while(scanf("%d",&a[n])!=EOF)n++;
    n--;
    dp[1]=a[1];
    for(int i=2;i<=n;i++){
        if(a[i]<=dp[j-1]){
        	dp[j]=a[i];
            j++;
		}
        else{
        	int k=1;
            while(dp[k]>=a[i]){
            	k++;
			}
            dp[k]=a[i];
        }
    }
    printf("%d\n",j-1);
    j=1;
    for(int i=1;i<=n;i++){
        if(i>1&&dp2[j-1]<a[i]){
            dp2[j]=a[i];
            j++;
        }
        else{
            int mid=(j-1)/2,l=0,r=j-1;
            while(l+1<r){
                if(a[i]>=dp2[mid]){
                    l=mid;
                    mid=(l+r)/2;
                }
                else if(a[i]<dp2[mid]){
                    r=mid;
                    mid=(l+r)/2;
                }
            }
            if(dp2[r-1]>a[i])r--;
            if(dp2[r-1]==a[i])continue;
            dp2[r]=a[i];
        }
    }
    printf("%d",j);
}
