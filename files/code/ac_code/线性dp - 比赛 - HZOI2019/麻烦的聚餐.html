#include <algorithm>
#include <cstdio>
#include <iostream>
using namespace std;
int a[30005], b[30005], dpa[30005], dpb[30005];
const int inf = 0x7f7f7f7f;
int main() {
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        dpa[i] = dpb[i]=inf;
        scanf("%d", &a[i]);
        b[n - i - 1] = a[i];
    }
    dpb[n]=dpa[n]=inf;
    for (int i = 0; i < n; i++) {
        *(upper_bound(dpa, dpa + n, a[i])) = a[i];
        *(upper_bound(dpb, dpb + n, b[i])) = b[i];
    }
    printf("%d", n - max(lower_bound(dpa, dpa + n, inf) - dpa,lower_bound(dpb, dpb + n, inf) - dpb));
    return 0;
}
