#include<stdio.h>

int main(){
	int n,a[105],dp[105]={0},dpi,dp2[105]={0},max=0;
	scanf("%d",&n);
	int i2=n;
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		dp[i]=1;
		dp2[i]=1;
		i2--;
	}
	
	for(int i=1;i<=n;i++){
		for(int j=1;j<i;j++){
			if(dp[j]+1>dp[i]&&a[i]>a[j]){
				dp[i]=dp[j]+1;
			}
		}
		if(dp[i]>dpi){
			dpi=dp[i];
		}
	}
	
	for(int i=n;i>=1;i--){
		for(int j=n;j>i;j--){
			if(dp2[j]+1>dp2[i]&&a[i]>a[j]){
				dp2[i]=dp2[j]+1;
			}
		}
	}
	for(int i=1;i<=n;i++){
		if(dp[i]+dp2[i]-1>max){
			max=dp[i]+dp2[i]-1;
		}
	}
	printf("%d",n-max);
}
