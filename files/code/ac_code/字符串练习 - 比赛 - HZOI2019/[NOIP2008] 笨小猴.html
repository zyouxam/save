#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
using namespace std;
int judge(int n){
        if(n==1 || n==0)return false;
        bool flag=true;
        for(int i=2;i<=sqrt(n);i++){
                if(n%i==0){
                        flag=false;
                        break;
                }
        }
        return flag;
}
int main(){
	char word[105]={0};
	int hash[130]={0};
	int maxn=0,minn=0x7fffffff;
	scanf("%s",word);
	for(int i=0;i<=105;i++){
	        if(!word[i])break;
	        hash[word[i]]++;
	}
	for(int i=0;i<130;i++){
	        maxn=max(maxn,hash[i]);
	        //printf("%d ",minn);
	        if(hash[i])minn=min(minn,hash[i]);
	}
	//printf("%d %d\n",maxn,minn);
	if(judge(maxn-minn)){
	        printf("Lucky Word\n%d",maxn-minn);
	}
	else{
	        printf("No Answer\n0");
	}
	return 0;
}
