#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int MAXN=2000;
void print(int da[]){
        bool zero=true;
        for(int i=MAXN-1;i>=0;i--){
                if(da[i])zero=false;
                if(!zero)printf("%d",da[i]);
        }
        if(zero)printf("0");
        printf("\n");
}
void add(int a[],int b[],int c[]){
        for(int i=0;i<MAXN;i++)c[i]=0;
        for(int i=0;i<MAXN;i++){
                c[i]+=b[i]+a[i];
                if(c[i]>9){
                        c[i+1]++;
                        c[i]-=10;
                }
        }
}
void dan_cheng(int a[],int b,int c[],int p=0){
        for(int i=0;i<MAXN;i++)c[i]=0;
        for(int i=0;i<MAXN;i++){
                c[i+p]+=b*a[i];
                if(c[i]>9){
                        c[i+1+p]+=c[i+p]/10;
                        c[i+p]%=10;
                }
        }
}
void cheng(int a[],int b[],int c[]){
        int d[MAXN]={0},e[MAXN]={0};
        for(int i=0;i<MAXN;i++){
                dan_cheng(a,b[i],d,i);
                for(int i=0;i<MAXN;i++)e[i]=c[i];
                add(d,e,c);
        }
}
int main(){
        char a[MAXN]={0},b[MAXN]={0};
	int da[MAXN]={0},db[MAXN]={0},dc[MAXN]={0};
        scanf("%s%s",a,b);
        int s1=strlen(a),s2=strlen(b);
        int j=0;
        for(int i=s1-1;i>=0;i--){
                da[j]=a[i]-'0';
                j++;
        }
        j=0;
        for(int i=s2-1;i>=0;i--){
                db[j]=b[i]-'0';
                j++;
        }
        cheng(da,db,dc);
        print(dc);
        return 0;
}
