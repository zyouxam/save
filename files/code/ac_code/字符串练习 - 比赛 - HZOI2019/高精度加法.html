#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int MAXN=250;
int main(){
        char a[MAXN]={0},b[MAXN]={0};
	int c[MAXN]={0};
        scanf("%s%s",a,b);
        int s1=strlen(a),s2=strlen(b),s=max(s1,s2);
        s++;
        if(s1<s){
                int i=s,j=s1;
                while(j>=0){
                        a[i]=a[j];
                        a[j]='0';
                        i--;
                        j--;
                }
                for(int i=0;i<s;i++){
                        if(!a[i])a[i]='0';
                }
        }
        if(s2<s){
                int i=s,j=s2;
                while(j>=0){
                        b[i]=b[j];
                        b[j]='0';
                        i--;
                        j--;
                }
                for(int i=0;i<s;i++){
                        if(!b[i])b[i]='0';
                }
        }
        //printf("\na:%s\nb:%s\n",a,b);
        for(int i=s;i>=1;i--){
                c[i]+=a[i]+b[i]-96;
                if(c[i]>=10){
                        c[i]-=10;
                        c[i-1]++;
                }
        }
        //printf("%s\n%s\n",a,b);
        for(int i=0;i<s;i++){
                if(!i && !c[i])continue;
                printf("%d",c[i]);
        }
        printf("\n");
        return 0;
}
