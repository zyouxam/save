#include<cstdio>
int a[100005]={0};
long long s=0,m,k;
bool check(long long num) {
    int id = 1;     
    long long tot = 0;  
    for (int i = 0; i < m; i ++) {  
        if (num - tot >= a[i]) {    
            tot += a[i];        
        }
        else if (a[i] > num)    
            return false;
        else {
            id ++;  
            tot = a[i]; 
            if (id > k) return false; 
        }
    }
    return true;    
}
int main(){
	freopen("maxmin.in","r",stdin);
	freopen("maxmin.out","w",stdout);
        scanf("%lld%lld",&m,&k);
        for(int i=0;i<m;i++){
                scanf("%d",&a[i]);
                s+=a[i];
        }
        long long l=0,r=s,mid;
        while(l<r-1){
                //printf("%lld (%lld) %lld [%d]\n",l,mid,r,check(mid));
                mid=(r+l)/2;
                if(check(mid))r=mid;
                else l=mid;
        }
        printf("%lld",r);
        return 0;
}
