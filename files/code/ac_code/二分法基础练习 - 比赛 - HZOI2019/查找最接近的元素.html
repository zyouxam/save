#include<cstdio>
long long a[100005]={0};
int main(){
	freopen("nearest.in","r",stdin);
	freopen("nearest.out","w",stdout);
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%lld",&a[i]);
	}
	
	int m;
	scanf("%d",&m);
	for(int i=0;i<m;i++){
		int c;
		scanf("%d",&c);
		int l=0,r=n-1,mid;
		while(l<r-1){
			mid=l+(r-l)/2;
			//mid=(r+l)/2;
			if(a[mid]<c){
				//printf("mid:%d l:%d r:%d\n",mid,l,r);
				l=mid;
			}
			else if(a[mid]>c){
				//printf("mid:%d l:%d r:%d\n",mid,l,r);
				r=mid;
			}
			else{
				break;
			}
		}
		//printf("mid:%d a[mid]:%lld a[mid+1]:%lld\n",mid,a[mid],a[mid+1]);
		if(a[mid]>c)mid-=1;
		if(mid<0)mid=0;
		else if(mid>=n)mid=n-1;
		if(c-a[mid]<=a[mid+1]-c)printf("%lld\n",a[mid]);
		else printf("%lld\n",a[mid+1]);
	
	}
	return 0;
}
