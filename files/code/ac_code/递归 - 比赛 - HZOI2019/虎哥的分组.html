#include<cstdio>
using namespace std;
long long s[301][301];
int main(){
    freopen("tiger.in","r",stdin);
    freopen("tiger.out","w",stdout);
    int n,k;
    scanf("%d%d",&n,&k);
    s[0][0] = 1;
    s[1][0] = 0;
    s[0][1] = 0;
    for(int i = 1;i <= n;i++){
        for(int j = 1;j <= k;j++){
            if(j > i)s[i][j] = 0;
            else s[i][j]=(j*(s[i-1][j]%9999991)+s[i-1][j-1]%9999991)%9999991;
        }
    }
    printf("%lld",s[n][k]%9999991);
    return 0;
 }
