3 a b c

#include<stdio.h>
int n;
void f(char from,int k,char to,char help){
	if(k==1){
		printf("%d:%c->%c\n",k,from,to);
		return;
	}
	f(from,k-1,help,to);
	printf("%d:%c->%c\n",k,from,to);
	f(help,k-1,to,from);
}
int main(){
freopen("hanoi.in","r",stdin);
	freopen("hanoi.out","w",stdout);
	char from,to,help;
	scanf("%d %c %c %c",&n,&from,&help,&to);
	f(from,n,to,help);
}
