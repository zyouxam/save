#include<cstdio>
int s[51][51]={0};
int f(int n,int m){
        if(n<1||m<1)return 0;
        if(n==1||m==1)return 1;
        if(s[n][m])return s[n][m];
        if(n<m)return s[n][m]=f(n,n);
        if(n==m)return s[n][m]=f(n,m-1)+1;
        return s[n][m]=f(n,m-1)+f(n-m,m);
}
int main(){
    freopen("seperate.in","r",stdin);
    freopen("seperate.out","w",stdout);
	int n,m;
	scanf("%d%d",&n,&m);
	printf("%d",f(n,m));
	return 0;
}
