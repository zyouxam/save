#include<cstdio>
int m,n;
int dx[4]={-2,2,-1,1};
int dy[4]={1,1,2,2};
int ans,visited[21][21];
void dfs(int x,int y){
        if(x==m && y==n){
                ans++;
                //printf("%d\n",ans);
                return;
        }
        for(int i=0;i<4;i++){
                int nx=x+dx[i],ny=y+dy[i];
                if(nx<=m && nx>=1 && ny>=1 && ny<=n && !visited[nx][ny]){
                        visited[nx][ny]=1;
                        //printf("(%d,%d)->(%d,%d)\n",x,y,nx,ny);
                        //getchar();
                        dfs(nx,ny);
                        visited[nx][ny]=0;
                }
        }
}
int main(){
		freopen("horse.in","r",stdin);
		freopen("horse.out","w",stdout);
        scanf("%d %d",&m,&n);
        visited[1][1]=1;
        dfs(1,1);
        printf("%d",ans);
        return 0;
}
