#include<cstdio>
#include<cstring>
int s2[51]={0};
int s4[51][51]={0};

int f1(int n,int m){
        if(n<1||m<1)return 0;
        if(n==1||m==1)return 1;
        if(n<m)return f1(n,n);
        if(n==m)return f1(n,m-1)+1;
        return f1(n,m-1)+f1(n-m,m);
}
int f4(int n,int m){
        if(n<1||m<1)return 0;
        if(n==1||m==1)return 1;
        if(s4[n][m])return s4[n][m];
        if(!(m%2))return s4[n][m]=f4(n,m-1);
        if(n<m)return s4[n][m]=f4(n,n);
        if(n==m)return s4[n][m]=f4(n,m-2)+1;
        return s4[n][m]=f4(n,m-2)+f4(n-m,m);
}
int f5(int n,int m){
        if(n>m && m==1)return 0;
        if(n<1||m<1)return 0;
        if(n==1||m==1)return 1;
        if(n<m)return f5(n,n);
        if(n==m)return f5(n,m-1)+1;
        return f5(n,m-1)+f5(n-m,m-1);
}
int f2(int n,int k){
	if (k==0||k>n) return 0;
	if (k==1||k==n) return 1;
	return f2(n-1,k-1)+f2(n-k,k);
}
int main(){
    freopen("seperate2.in","r",stdin);
    freopen("seperate2.out","w",stdout);
	int n,k;
	scanf("%d%d",&n,&k);
	printf("%d\n",f1(n,n));//1
	printf("%d\n",f2(n,k));//2----
	printf("%d\n",f1(n,k));//3
	printf("%d\n",f4(n,n));//4
	printf("%d\n",f5(n,n));//5
	return 0;
}
