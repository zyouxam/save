#include<cstdio>
#include<cmath>
using namespace std;
int n,p,q,d[300001];
char c;
int read() {
	int num = 0, flag = 1; char c = getchar();
	for(; c < '0' || c > '9'; c = getchar())
		if(c == '-') flag = 0;
	for(; c >= '0' && c <= '9'; c = getchar())
		num = (num<<3)+(num<<1)+c-48;
	return flag? num: -num;
}
int fa[300010],size[300010];
int getfather(int x) {
	if(x==fa[x]) return x;
	int root=getfather(fa[x]);
	d[x]+=d[fa[x]];
	return fa[x]=root;
}
void merge(int x,int y) {
	x=getfather(x),y=getfather(y);
	fa[x]=y;d[x]=size[y];
	size[y]+=size[x];
}

int main() {
	n=read();
	for(int i=1;i<=300000;i++) fa[i]=i,size[i]=1;
	for(int i=1;i<=n;i++) {
		c=getchar();p=read();q=read();
		if(c=='M') merge(p,q); 
		if(c=='C') {
			if(getfather(p)!=getfather(q)) puts("-1\n");
			else {
				int ans=abs(d[p]-d[q])+1;
				printf("%d\n",ans-2);
			}
		}
	}
	return 0;
}
