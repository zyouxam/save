#include<cstdio>
using namespace std;
int n,a[1001][1001],f[1001],pd[1001];
int find(int x)//并查集
{
    return f[x]==x?x:f[x]=find(f[x]);
}
int main()
{
    scanf("%d",&n);//读入
    for (int i=1;i<=n;i++)
    {
        f[i]=i;//初始化
        pd[i]=1;//初始化
        scanf("%d",&a[i][0]);//读入
        for (int j=1;j<=a[i][0];j++)
        scanf("%d",&a[i][j]);//读入
    }
    for (int hh=n;hh>=1;hh--)
    {
        for (int i=1;i<=a[hh][0];i++)
        if (find(hh)!=find(a[hh][i])&&a[hh][i]>=hh)//判断之前是否已经连接过，或是后面会连接
        {//连接这两个点
            int xx=find(hh),yy=find(a[hh][i]);
            if (xx<yy)
            {
                f[xx]=yy;
                pd[yy]+=pd[xx];//危险程度累计
            }
            else
            {
                f[yy]=xx;
                pd[xx]+=pd[yy];//危险程度累计
            }
        }
        for (int i=1;i<=n;i++)
        if (pd[i]>n/2)//判断危险程度是否超过n/2
        {
            printf("%d",hh);//输出
            return 0;//直接退出
        }
    }
    return 0;
}
