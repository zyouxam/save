using namespace std;
map<string,string> f;
string find(string x){
    return f[x]==x?x:(f[x]=find(f[x]));
}
void join(string a,string b){
    //cout<<2<<find(a)<<" "<<find(b)<<endl;
    f[find(a)]=find(b);
    //cout<<3<<find(a)<<" "<<find(b)<<endl;
}
int main(){
    string cmd;
    string cfather;
    do{
        cin>>cmd;
        string name=cmd.substr(1,cmd.length()-1);
        if(cmd[0]=='#'){
            cfather=name;
            if(!f.count(name))f[name]=name;
        }else if(cmd[0]=='+'){
            if(!f.count(name))f[name]=name;
            //cout<<1<<cfather<<" "<<name<<endl;
            join(name,cfather);
            //cout<<find(name)<<endl;
        }else if(cmd[0]=='?'){
            cout<<name<<" "<<find(name)<<endl;;
        }
    }while(cmd!="$");
}
