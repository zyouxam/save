#include <cstdio>
#include<cmath>
#define MAXN 1005
int f[MAXN];
long long n,data[MAXN][3];
int find(int x){
    return f[x]==x?x:(f[x]=find(f[x]));
}
void join(int a,int b){
    //printf("%d %d joined\n",a,b);
    f[find(a)]=find(b);
}
double dis(long long a,long long b){
    return sqrt((data[a][0]-data[b][0])*(data[a][0]-data[b][0])+(data[a][1]-data[b][1])*(data[a][1]-data[b][1])+(data[a][2]-data[b][2])*(data[a][2]-data[b][2]));
}
int main(){
    int zn;
    scanf("%d",&zn);
    while(zn--){
        int n,h,r;
        scanf("%d%d%d",&n,&h,&r);
        for(int i=1;i<=n+2;i++)f[i]=i;
        for(int i=1;i<=n;i++)scanf("%lld%lld%lld",&data[i][0],&data[i][1],&data[i][2]);
        for(int i=1;i<=n;i++){
            if(data[i][2]<=r)join(i,n+2);
            if(h-data[i][2]<=r)join(i,n+1);
            for(int j=1;j<=n;j++){
                if(i==j)continue;
                if(dis(i,j)<=2*r)join(i,j);
            }
        }
        //for(int i=1;i<=n+2;i++)printf("%d ",f[i]);
        printf("%s\n",find(n+1)==find(n+2)?"Yes":"No");
    }
    return 0;
}
