#include<cstdio>
#include<algorithm>
#define MAXN 20005
using namespace std;
int f[MAXN];
int find(int x){
    if(f[x]==x)return x;
    return find(f[x]);
}
void join(int a,int b){
    f[find(a)]=find(b);
}
int main(){
    int n,a,b,m,q;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)f[i]=i;
    for(int i=1;i<=m;i++){
        scanf("%d%d",&a,&b);
        join(a,b);
    }
    scanf("%d",&q);
    for(int i=1;i<=q;i++){
        scanf("%d%d",&a,&b);
        printf("%s\n",find(a)==find(b)?"Yes":"No");
    }
    return 0;
}
