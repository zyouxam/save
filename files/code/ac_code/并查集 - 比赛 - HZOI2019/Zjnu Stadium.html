#include <cstdio>
#include <iostream>
using namespace std;
#define MAXN 50010
int n, m, father[MAXN], way[MAXN], ans;
int find(int x) {
    if (x == father[x]) return father[x];
    int fa = father[x];
    father[x] = find(father[x]);
    way[x] += way[fa];
    return father[x];
}
void join(int x, int y, int x1, int y1, int z) {
    father[y1] = x1;
    way[y1] = way[x] + z - way[y];
}
int main() {
    while (~scanf("%d%d", &n, &m)) {
        for (int i = 0; i < MAXN; i++) father[i] = i, way[i] = 0;
        ans = 0;
        int x, y, z;
        while (m--) {
            scanf("%d%d%d", &x, &y, &z);
            int f1 = find(x), f2 = find(y);
            if (f1 != f2)
                join(x, y, f1, f2, z);
            else if (way[y] - way[x] != z)
                ans++;
        }
        printf("%d\n", ans);
    }
}
