#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int MAXN=1000;
void print(int da[]){
        bool zero=true;
        for(int i=MAXN-1;i>=0;i--){
                if(da[i])zero=false;
                if(!zero)printf("%d",da[i]);
        }
        if(zero)printf("0");
        printf("\n");
}
void add(int a[],int b[],int c[]){
        for(int i=0;i<MAXN;i++){
                c[i]+=b[i]+a[i];
                if(c[i]>9){
                        c[i+1]++;
                        c[i]-=10;
                }
        }
}
int main(){
        int n,a[MAXN]={1},b[MAXN]={1},c[MAXN]={0},d[MAXN]={0};
        scanf("%d",&n);
        for(int i=1;i<n;i++){
                //c=b;
                for(int j=0;j<MAXN;j++)c[j]=b[j];
                //b=a+b;
                for(int j=0;j<MAXN;j++)d[j]=0;
                add(a,b,d);
                for(int j=0;j<MAXN;j++)b[j]=d[j];
                
                //a=c;
                for(int j=0;j<MAXN;j++)a[j]=c[j];
        }
        print(a);
        return 0;
}
