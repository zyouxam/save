#include<cstdio>
#include<cstdlib>
char up(char a){
        if(a>='a' and a<='z')return a-32;
        else return a;
}
char lo(char a){
        if(a>='A' and a<='Z')return a+32;
        else return a;
}
int main(){
        int p1,p2,p3;
        char s[105]={0};
        scanf("%d %d %d",&p1,&p2,&p3);
        scanf("%s",s);
	system("echo 1>in");
        for(int i=0;i<105;i++){
                if(!s[i])break;
                char begin=s[i-1],end=s[i+1];
                if(s[i]!='-' || begin>=end || !(lo(begin)>='a' && lo(end)<='z' || lo(begin)>='0' && lo(end)<='9')){
                        printf("%c",s[i]);
                        continue;
                }
                char z[1000]={0};
                int zi=0;
                for(int j=begin+1;j<=end-1;j++){
                        if(p1==1)for(int k=0;k<p2;k++)z[zi++]=lo(j);
                        else if(p1==2)for(int k=0;k<p2;k++)z[zi++]=up(j);
                        else if(p1==3)for(int k=0;k<p2;k++)z[zi++]='*';
                }
                if(p3==1){
                        for(int j=0;j<zi;j++)printf("%c",z[j]);
                }
                else if(p3==2){
                        for(int j=zi-1;j>=0;j--)printf("%c",z[j]);
                }
        }
        printf("\n");
        return 0;
}
