#include<cstdio>
#include<cmath>
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
	        int k;
	        scanf("%d",&k);
	        bool isZhi = true;
                for(int j=2;j<=sqrt(k);j++){
                        if(k%j==0){
                                isZhi=false;
                                printf("NO\n");
                                break;
                        }
                }
                if(isZhi)printf("YES\n");
	}
	return 0;
}
