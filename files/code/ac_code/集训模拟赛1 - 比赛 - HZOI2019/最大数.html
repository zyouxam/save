#include<cstdio>
#include<iostream>
#define ull unsigned long long
using namespace std;
ull tree[800005], arg[200005];
char opt[200005];
ull query(int p, int l, int r, int s, int t) {
    if(s<=l && r<=t) return tree[p];
    int m=(l+r)>>1;
    ull ans=0;
    if(s<=m) ans=max(ans, query(p<<1, l, m, s, t));
    if(t>m) ans=max(ans, query(p<<1|1, m+1, r, s, t));
    return ans;
}
void update(int p, int l, int r, int t, ull v){
    if(l==r) return tree[p]=v, void();
    int m=(l+r)>>1;
    if(t<=m) update(p<<1, l, m, t, v);
    else update(p<<1|1, m+1, r, t, v);
    tree[p]=max(tree[p<<1], tree[p<<1|1]);
}
int main() {
    int m, mi=0, n=0;
    ull t=0, d;
    scanf("%d%llu", &m, &d);
    for(int i=1;i<=m;i++){
        scanf(" %c%llu", &opt[i], &arg[i]);
        if(opt[i]=='A')n++;
    }
    for(int i=1;i<=m;i++){
        if(opt[i]=='Q') {
            printf("%llu\n", t=query(1, 1, n, mi-arg[i]+1, mi));
        } else {
            mi++;
            update(1, 1, n, mi, (arg[i]+t)%d);
        }
    }
    return 0;
}
