#include <bits/stdc++.h>
#define N 10
#define LL long long
using namespace std;
LL f[1<<N][1<<N], ans;
char s[15];
int a[15], cnt[15];
int main(){
	int i, j, k, n, m, T, d;
	scanf("%d", &T);
	while(T--){
		memset(f, 0, sizeof(f));
		memset(cnt, 0, sizeof(cnt));
		scanf("%s%d", s, &d);
		n = strlen(s);
		for(i = 0; i < n; i++){
			a[i] = s[i] - 48;
			cnt[a[i]]++;
			f[1<<i][a[i]%d] = 1;
		}
		for(i = 0; i < (1 << n); i++){
			for(j = 0; j < d; j++){
				for(k = 0; k < n; k++){
					if(!(1 << k & i)) f[i | (1 << k)][(j * 10 + a[k]) % d] += f[i][j];
				}
			}
		}
		ans = f[(1 << n) - 1][0];
		for(i = 0; i < 10; i++){
			for(j = 2; j <= cnt[i]; j++) ans /= j;
		}
		printf("%d\n", ans);
	}
	return 0;
}
