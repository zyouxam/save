#include <bits/stdc++.h>
using namespace std;
const int maxn = 500000 + 10;
inline long long gi() {
    char ch = getchar();
    long long x = 0, q = 0;
    while (ch < '0' || ch > '9') ch == '-' ? q = 1 : 0, ch = getchar();
    while (ch >= '0' && ch <= '9') x = x * 10 + ch - '0', ch = getchar();
    return q ? (-x) : x;
}
#define mid ((l + r) >> 1)
#define lson rt << 1, l, mid
#define rson rt << 1 | 1, mid + 1, r
#define len (r - l + 1)
long long n, m;
struct Edge { long long next, to; } h[maxn << 1];
long long head[maxn << 1];
long long tot, to[maxn], w[maxn], wt[maxn];
long long a[maxn << 2], laz[maxn << 2];
long long son[maxn], id[maxn], fa[maxn], cnt, dep[maxn], siz[maxn], top[maxn];
long long res = 0;
void Add(long long u, long long v) {
    h[++tot].next = head[u];
    h[tot].to = v;
    head[u] = tot;
}
void pushdown(long long rt, long long lenn) {
    laz[rt << 1] += laz[rt];
    laz[rt << 1 | 1] += laz[rt];
    a[rt << 1] += laz[rt] * (lenn - (lenn >> 1));
    a[rt << 1 | 1] += laz[rt] * (lenn >> 1);
    laz[rt] = 0;
}
void build(long long rt, long long l, long long r) {
    if (l == r) return a[rt] = wt[l], void();
    build(lson), build(rson);
    a[rt] = (a[rt << 1] + a[rt << 1 | 1]);
}
void query(long long rt, long long l, long long r, long long L, long long R) {
    if (L <= l && r <= R) return res += a[rt], void();
    else {
        if (laz[rt]) pushdown(rt, len);
        if (L <= mid) query(lson, L, R);
        if (R > mid) query(rson, L, R);
    }
}
void update(long long rt, long long l, long long r, long long L, long long R, long long k) {
    if (L <= l && r <= R) {
        laz[rt] += k;
        a[rt] += k * len;
    } else {
        if (laz[rt]) pushdown(rt, len);
        if (L <= mid) update(lson, L, R, k);
        if (R > mid) update(rson, L, R, k);
        a[rt] = (a[rt << 1] + a[rt << 1 | 1]);
    }
}
long long qRange(long long x, long long y) {
    long long ans = 0;
    while (top[x] != top[y]) {
        if (dep[top[x]] < dep[top[y]]) swap(x, y);
        res = 0;
        query(1, 1, n, id[top[x]], id[x]);
        ans += res;
        x = fa[top[x]];
    }
    if (dep[x] > dep[y]) swap(x, y);
    res = 0;
    query(1, 1, n, id[x], id[y]);
    ans += res;
    return ans;
}
void updSon(long long x, long long k) {
    update(1, 1, n, id[x], id[x] + siz[x] - 1, k);
}

void dfs1(long long x, long long f, long long deep) {
    dep[x] = deep;
    fa[x] = f;
    siz[x] = 1;
    long long maxson = -1;
    for (long long i = head[x]; i; i = h[i].next) {
        long long y = h[i].to;
        if (y == f) continue;
        dfs1(y, x, deep + 1);
        siz[x] += siz[y];
        if (siz[y] > maxson) son[x] = y, maxson = siz[y];
    }
}
void dfs2(long long x, long long topf) {
    id[x] = ++cnt;
    wt[cnt] = w[x];
    top[x] = topf;
    if (!son[x]) return;
    dfs2(son[x], topf);
    for (long long i = head[x]; i; i = h[i].next) {
        long long y = h[i].to;
        if (y == fa[x] || y == son[x]) continue;
        dfs2(y, y);
    }
}
int main() {
    n = gi(), m = gi();
    for (long long i = 1; i <= n; i++) w[i] = gi();
    for (long long i = 1; i < n; i++) {
        long long a, b;
        a = gi(), b = gi();
        Add(a, b), Add(b, a);
    }
    dfs1(1, 0, 1), dfs2(1, 1);
    build(1, 1, n);
    while (m--) {
        long long k, x, y, z;
        k = gi();
        if (k == 1) {
            x = gi(), y = gi();
            update(1, 1, n, id[x], id[x], y);
        } else if (k == 2) {
            x = gi(), y = gi();
            updSon(x, y);
        } else {
            x = gi();
            printf("%lld\n", qRange(1, x));
        }
    }
    return 0;
}
