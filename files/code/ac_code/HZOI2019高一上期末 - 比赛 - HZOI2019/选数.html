#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int ans;
int a[25];
int n;
int ps[100000];
bool isp(int n){
	bool ind=(n<100000&&n>=0);
	if(n<2)return false;
	if(ind){
		if(ps[n]==1)return true;
		else if(ps[n]==2)return false;
	}
	for(int i=2;i<=sqrt(n);i++){
		if(!(n%i)){
			if(ind)ps[n]=2;
			return false;
		}
	}
	if(ind)ps[n]=1;
	return true;
}
void dfs(int k,int h,int zi){
	if(k<=0){
		//printf("[%d]\n",h);
		if(isp(h))ans++;
		return;
	}
	if(zi>=n)return;
	for(int i=zi;i<n;i++){
		dfs(k-1,h+a[i],i+1);
	}
}
int main(){
	freopen("choose.in","r",stdin);
	freopen("choose.out","w",stdout);
	int k;
	scanf("%d%d",&n,&k);
	for (int i = 0; i < n; i++){
		scanf("%d",&a[i]);
	}
	dfs(k,0,0);
	printf("%d",ans);
	
	return 0;
}
