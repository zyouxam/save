#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define MAXH 45005
#define MAXN 505
int s[MAXN],f[MAXH];
int h,n;
int main(){
	freopen("weight.in","r",stdin);
	freopen("weight.out","w",stdout);
	scanf("%d %d",&h,&n);
	for (int i = 0; i < n; i++){
		scanf("%d",&s[i]);
	}
	for (int i = 0; i < n; i++){
		for (int j = h; j >= s[i]; j--){
			f[j]=max(f[j],f[j-s[i]]+s[i]);
		}
	}
	printf("%d",f[h]);
	
	return 0;
}
