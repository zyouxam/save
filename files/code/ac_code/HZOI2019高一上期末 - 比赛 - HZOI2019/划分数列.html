#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
long long n,k,a[100005];
bool check(long long maxh){
	long long i=0,nsum=0,sepai=1;
	while(i<n&&sepai<=k){
		
		if(nsum+a[i]>maxh){
			sepai++;
			nsum=0;
		} else {
			nsum+=a[i++];
		}
	}
	
	if(sepai>k)return false;
	else return true;
}
int main(){
	freopen("sepa.in","r",stdin);
	freopen("sepa.out","w",stdout);
	
	scanf("%lld%lld",&n,&k);
	long long sum=0;
	for (int i = 0; i < n; i++){
		scanf("%lld",&a[i]);
		sum+=a[i];
	}
	long long l=0,r=sum,mid;
	//printf("\n%d",sum);
	while(l+1<r){
		mid=(l+r)/2;
		
		if(check(mid)){
			r=mid;
		}else{
			l=mid;
		}
	}	
	printf("%lld",r);
	
	return 0;
}
