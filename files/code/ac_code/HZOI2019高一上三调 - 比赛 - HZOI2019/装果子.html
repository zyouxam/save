#include<cstdio>
long long a[100005],n,m,sum;
bool check(long long v){
    long long i=0,lv=v,tot=1;
    while(i<n){
        if(a[i]>v)return false;//v小于某树下果子体积，v不能取
        if(a[i]<=lv){
            lv-=a[i];//如果能盛得下，lv减去当前果子体积
            i++;
        }else{
            tot++;//使用袋子数量+1
            if(tot>m)return false;
            lv=v;//当前袋子剩余体积为v
        }
    }
    return true;
}
int main(){
    freopen("fruit.in","r",stdin);
    freopen("fruit.out","w",stdout);
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++){
        scanf("%lld",&a[i]);
        sum+=a[i];
    }
    long long l=1,r=sum,mid;
    while(l<r-1){
        mid=(l+r)/2;
        if(check(mid))r=mid;
        else l=mid;
    }
    printf("%lld",r);
    return 0;
}
