#include<cstdio>
#include<algorithm>
using namespace std;
int a[200005];
int main(){
    freopen("simple.in","r",stdin);
    freopen("simple.out","w",stdout);
    long long n,s=0;
    scanf("%lld",&n);
    for(long long i=0;i<2*n;i++){
        scanf("%d",&a[i]);
    }
    sort(a,a+n+n);
    for(long long i=0;i<2*n;i+=2){
        s+=a[i];
    }
    printf("%lld",s);
    return 0;
}
