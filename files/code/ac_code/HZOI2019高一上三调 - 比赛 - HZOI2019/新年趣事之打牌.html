#include <cstdio>
using namespace std;
const int maxn=100050;
int Total_W,N;
int dp[maxn],father[maxn],input[200];
bool used[200];
int main(){
	freopen("card.in","r",stdin);
    freopen("card.out","w",stdout);
    dp[0]=1;
    scanf("%d%d",&Total_W,&N);
    if (Total_W==0)  {for (int n=1;n<=N;n++) printf("%d ",n);return 0;}
    for (int n=1;n<=N;n++) scanf("%d",&input[n]);
    for (int n=1;n<=N;n++)
        for (int m=Total_W;m>=input[n];m--){
            if (dp[m-input[n]]==1){
                if (!dp[m]){
                    dp[m]=1;
                    father[m]=n;
                }
                else dp[m]=-1;
            }
            else if (dp[m-input[n]]==-1) dp[m]=-1;
        }
    if (!dp[Total_W]||dp[Total_W]==-1) {printf("%d",dp[Total_W]);return 0;}
    int s=Total_W;

    while (s!=0){
        used[father[s]]=1;
        s-=input[father[s]];
    }
    for (int n=1;n<=N;n++)if (!used[n]) printf("%d ",n);
    return 0;

}
