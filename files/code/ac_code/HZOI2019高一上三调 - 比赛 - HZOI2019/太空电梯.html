#include<bits/stdc++.h>
using namespace std;
struct material{
	int h,limit,num;
};
bool cmp(material a,material b){
	return a.limit<b.limit;
}
material all[500];
int f[9999999];
void solve(int k,int maxlim,int minlim){
	for (int i=1;i<=k;i++){
		int tot=0;
		for (int j=1;tot+j<=all[i].num&&j*all[i].h<=all[i].limit;j<<=1){
			tot+=j;
			for (int l=all[i].limit ; l>=j*all[i].h ; l--)
				f[l]=max(f[l],f[l-j*all[i].h]+j*all[i].h);
		}
		int x=all[i].num-tot;
		if(x){
			for (int j=all[i].limit ; j>=x*all[i].h ; j--)
				f[j]=max(f[j],f[j-x*all[i].h]+x*all[i].h);
		}	
	}
	int ans=-1;
	for (int i=minlim ;  i<=maxlim ; i++)
		ans=max(ans,f[i]);
	cout<<ans;
}
int main(){
	freopen("space.in","r",stdin);
	freopen("space.out","w",stdout);
	int k;
	cin>>k;
	int maxlim=-1,minlim=40005;
	for (int i=1;i<=k;i++){
		cin>>all[i].h>>all[i].limit>>all[i].num;
		maxlim=max(maxlim,all[i].limit);
		minlim=min(minlim,all[i].limit);
	}
	sort(all+1,all+1+k,cmp);
	solve(k,maxlim,minlim);
	return 0;
}
