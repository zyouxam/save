#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
struct Edge{
    int qs, zz, w;
    bool operator<(Edge b)const{return w < b.w;}
} E[100000];
long long f[1000], ans = 0, k = 0, n, m, w[1000], jl[1000];
int find(int x){return x == f[x] ? x : (f[x] = find(f[x]));}
void join(int x, int y){f[find(x)] = find(y);}
int main(){
    cin >> n;
    for (int i = 1; i <= n; ++i)scanf("%lld", &w[i]);
    for (int i = 1; i <= n + 1; ++i)f[i] = i;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= n; ++j){
            int x;
            scanf("%d", &x);
            if (x != 0){
                E[++m].qs = i;
                E[m].zz = j;
                E[m].w = x;
            }
        }
    for (int i = 1; i <= n; ++i){
        E[++m].qs = i;
        E[m].zz = n + 1;
        E[m].w = w[i];
        E[++m].qs = n + 1;
        E[m].zz = i;
        E[m].w = w[i];
    }
    sort(E + 1, E + m + 1);
    int k = 0;
    for (int j = 1; j <= m; ++j){
        if (find(E[j].qs) != find(E[j].zz)){
            join(E[j].qs, E[j].zz);
            ans += E[j].w;
            k++;
        }
        if (k == n)break;
    }
    cout << ans;
    return 0;
}
