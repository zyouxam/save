#include<cstdio>
#include<cstring>
#include<queue>
#define w first
#define node second
#define MAXN 2005
#define MAXM 4020024
#define INF 1e300
#define P pair<double,int>
using namespace std;
struct Edge{
    int to;
    double w;
    int next;
    Edge(){to=w=next=0;}
} edge[MAXM];
int head[MAXM];
int si;

double dis[MAXN];
bool vis[MAXN];
int main(){
    for(int i=0;i<MAXN;i++)dis[i]=INF,vis[i]=false;
    int n,m;
    scanf("%d%d",&n,&m);
    int x,y,z;
    for(int i=1;i<=m;i++){
        scanf("%d%d%d",&x,&y,&z);
        edge[++si].next=head[x];
        edge[si].to=y;
        edge[si].w=1-(double(z)/100);
        head[x]=si;

        edge[++si].next=head[y];
        edge[si].to=x;
        edge[si].w=1-(double(z)/100);
        head[y]=si;
    }
    int a,b;
    scanf("%d%d",&a,&b);
    dis[a]=1;
    priority_queue<P> q;
    q.push(make_pair(0,a));
    while(!q.empty()){
        P now=q.top();
        q.pop();
        if(vis[now.node])continue;
        vis[now.node]=true;
        for(int x=head[now.node];x;x=edge[x].next){
            if(dis[now.node]*edge[x].w>dis[edge[x].to] || dis[edge[x].to]==INF){
                //printf("%lf %lf %lf\n",dis[now.node],edge[x].w,dis[edge[x].to]);
                dis[edge[x].to]=dis[now.node]*edge[x].w;
                q.push(make_pair(dis[edge[x].to],edge[x].to));
            }
        }
    }
    printf("%.8lf",100/dis[b]);
}
