#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,m,x,y,v,f[301][301],map[301][301],t;
int main(){
    scanf("%d%d%d",&n,&m,&t);
    memset(f,127/3,sizeof(f));
    for (int i=1;i<=m;i++){
      scanf("%d%d%d",&x,&y,&v);
      f[x][y]=v;
    }
   for (int k=1;k<=n;k++)
     for (int i=1;i<=n;i++)
       for (int j=1;j<=n;j++)
         {
           if (i!=j&&j!=k&&i!=k)
            f[i][j]=min(f[i][j],max(f[i][k],f[k][j]));       
          }
   for (int i=1;i<=t;i++){
     scanf("%d%d",&x,&y);
     if (f[x][y]!=707406378) printf("%d\n",f[x][y]);
     else printf("-1\n");
   }
    
}
