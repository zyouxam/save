#include<stdio.h>
#include<iostream>
using namespace std;
int a[105][4];
int d[105][105];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			d[i][j]=10000000; 
		}
	}
	for(int i=1;i<=n;i++){
		//    人数      左节点  右节点   父节点 
		cin>>a[i][0]>>a[i][1]>>a[i][2];  
		//当前节点到左右节点的距离为1
		d[i][a[i][1]]=d[i][a[i][2]]=1;
		//左节点的父节点为当前节点 
		a[a[i][1]][3]=i;
		//右节点的父节点为当前节点 
		a[a[i][2]][3]=i;
		//当前节点 到 当前节点的父节点 距离为 1
		d[i][a[i][3]]=1;
	}
	/*
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			cout<<d[i][j]<<" ";
		}
		cout<<endl;
	}*/
	for(int k=1;k<=n;k++){
		for(int i=1;i<=n;i++){
			for(int j=1;j<=n;j++){
				if(i==j)d[i][j]=0;
				//if(a[i][3]&&a[j][3]&&a[k][3])
				else d[i][j]=min(d[i][j],d[i][k]+d[k][j]);
			}
		}
	}
	/* 
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			cout<<d[i][j]<<" ";
		}
		cout<<endl;
	}
	*/ 
	int min=2000000000;
	for(int i=1;i<=n;i++){
		int s=0;
		if(a[i][0])
		for(int j=1;j<=n;j++){
			if(a[j][0]){
				//cout<<d[i][j]*a[i][0]<<endl;
				s+=d[i][j]*a[j][0];
			}
		}
		//cout<<i<<" "<<s<<endl;
		if(s<min)min=s;
	}
	cout<<min;
}
