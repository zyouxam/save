#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
using namespace std;
int totnum, tot, fa[502];
int n, m, x, y, q[502], kl;
bool b[502];
struct dd {
    int kaishi, zhong, juli;
    int id;
} jie[5000];
int comp1(const dd &a, const dd &b) {
    if (a.juli == b.juli) return a.kaishi < b.kaishi;
    return a.juli < b.juli;
}
int find(int x) {
    if (fa[x] == x) return x;
    fa[x] = find(fa[x]);
    return fa[x];
}
int in() {
    char c = getchar();
    int x = 0;
    while (c < '0' || c > '9') c = getchar();
    for (; c >= '0' && c <= '9'; c = getchar()) x = x * 10 + c - '0';
    return x;
}
bool KUR(int t) {
    sort(jie + 1, jie + t + 1, comp1);
    int num = 0;
    totnum = 0;
    kl = 0;
    memset(b, 0, sizeof(b));
    for (int j = 1; j <= n; ++j) fa[j] = j;
    for (int j = 1; j <= t; ++j) {
        int hi = jie[j].kaishi;
        int hj = jie[j].zhong;
        int yu = find(hj);
        int lin = find(hi);
        if (yu != lin) {
            fa[yu] = lin;
            totnum += jie[j].juli;
            num++;
            q[++kl] = jie[j].id;
        }
        if (num == n - 1) return true;
    }
    return false;
}
int main() {
    n = in();
    m = in();
    for (int i = 1; i <= m; ++i) {
        jie[i].kaishi = in();
        jie[i].zhong = in();
        jie[i].juli = in();
        jie[i].id = i;
        if (i < n - 1) {
            printf("0\n");
            continue;
        } else {
            if (!KUR(i)) {
                printf("0\n");
                continue;
            } else {
                double fg = (double)totnum / 2;
                printf("%.1lf\n", fg);
            }
        }
    }
}
