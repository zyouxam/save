#include <cstdio>
#include<iostream>
#include<algorithm>
#define MAXM 20005
using namespace std;
struct edge{
    int a,b,w;
    bool operator<(edge b)const{return w<b.w;}
} edges[MAXM];
int ans;
int f[MAXM];
int find(int x){return f[x]==x?x:(f[x]=find(f[x]));}
void join(int a,int b,int w){f[find(a)]=find(b);ans+=w;}
int main(){
    int n,k,t1,t2,t3,t4,tot=0,bt=0;
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)f[i]=i;
    for(int i=1;i<=k;i++){
        scanf("%d%d%d%d",&t1,&t2,&t3,&t4);
        if(t1==2){
            edges[++tot].a=t2;
            edges[tot].b=t3;
            edges[tot].w=t4;
        }else{
            join(t2,t3,t4);
            bt++;
        }
    }
    int ti=0;
    sort(edges+1,edges+tot+1);
    for(int i=1;i<=tot;i++){
        int a=edges[i].a,b=edges[i].b,w=edges[i].w;
        if(find(a)==find(b))continue;
        //printf("[%d %d]\n",a,b);
        join(a,b,w);
        ti++;
    }
    printf("%d",ans);
}
