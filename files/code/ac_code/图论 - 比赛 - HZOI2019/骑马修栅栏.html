#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#define maxn 2001
using namespace std;
int g[maxn][maxn],du[maxn],path[maxn],n,m,tot,x,y,start;
void find(int i){
	//printf("%d\n",i);
	for(int j=1;j<=n;j++){
		if(g[i][j] && i!=j){
			g[i][j]--;
			g[j][i]--;
			find(j);
		}
	}
	path[tot++]=i;
}
int main(){
	freopen("fence.in","r",stdin);
	freopen("fence.out","w",stdout);
	scanf("%d",&m);
	for (int i = 0; i < m; i++){
		scanf("%d%d",&x,&y);
		g[x][y]++;
		g[y][x]++;
		du[x]++;
		du[y]++;
		n=max(n,max(x,y));
	}
	start=1;
	for (int i = 1; i <= n; i++){
		if(du[i]%2){
			start=i;
			break;
		}
	}
	find(start);
	for(int i=tot-1;i>=0;i--){
		printf("%d\n",path[i]);
	}
	return 0;
}
