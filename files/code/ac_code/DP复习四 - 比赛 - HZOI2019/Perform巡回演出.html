#include<cstdio>
#include<queue>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=1e3+100;
const long long INF=0x3f3f3f3f3f3f3f3f;
struct Edge{ int next,to;long long val; }edges[maxn*maxn];
int head[maxn*maxn], tot, vis[maxn*maxn], n, m;
void add(int x,int y,long long v){ edges[++tot]=(Edge){head[x], y, v}, head[x]=tot; }
long long dis[15][15][maxn],t[maxn][maxn],dist[maxn*maxn];
struct Node{
	int id; long long val;
	bool operator <(const Node&A)const{ return val>A.val; }
} u;
void spfa(){
	priority_queue<Node > q;
	memset(vis,0,sizeof(vis));
	memset(dist,0x3f,sizeof(dist));
	dist[1]=0;
	q.push((Node){1,0});
	while(!q.empty()){
		u=q.top(), q.pop();
		if(vis[u.id])continue;
		vis[u.id]=1;
		for(int i=head[u.id];i;i=edges[i].next){
			#define v edges[i].to
			if(dist[v]>dist[u.id]+edges[i].val)
				dist[v]=dist[u.id]+edges[i].val, q.push((Node){v,dist[v]});
			#undef v
		}
	}
}
int main(){
	while(~scanf("%d%d",&n,&m)&&n&&m){
		memset(head,0,sizeof(head));tot=0;
		for(int i=1;i<=n;i++){
			for(int j=1;j<=n;j++){
				if(i==j)continue;
				scanf("%lld",&t[i][j]);
				for(int k=0;k<t[i][j];k++){
					scanf("%lld",&dis[i][j][k]);
					if(dis[i][j][k]==0)
						dis[i][j][k]=INF;
				}
			}
		}
		for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++){
			for(int k=0;k<m;k++){
				if(i==j)continue;
				add(i+k*m,j+(k+1)*m,dis[i][j][k%t[i][j]]);
			}
		}
		spfa();
		printf("%lld\n",dist[n+m*m]==INF?0:dist[n+m*m]);
	}
}
