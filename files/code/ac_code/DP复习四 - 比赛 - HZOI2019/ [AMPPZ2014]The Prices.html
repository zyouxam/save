#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
inline int read(int f=1, int x=0, char c=getchar()){
    while(c<'0'||c>'9')f=(c=='-')?-1:1, c=getchar();
    while(c>='0'&&c<='9')x=(x<<1)+(x<<3)+(c^48), c=getchar();
    return f*x;
}
int d[101], store[101][17], f[101][65536];
int main() {
    int n=read(), m=read();
    for(int i=1;(i<=n)?(d[i]=read(),1):0;i++)
        for(int j=0;j<m;j++) store[i][j]=read();
    memset(f, 0x3f, sizeof(f));
    f[0][0]=0;
    for(int i=1;i<=n;i++){
        for(int s=0;s<(1<<m);s++){
            f[i][s]=f[i-1][s]+d[i];
            for(int k=0;k<m;k++) {
                if((s>>k)&1)
                    f[i][s]=min(f[i][s], f[i][s^(1<<k)]+store[i][k]);
            }
        }
        for(int s=0;s<(1<<m);s++){
            f[i][s]=min(f[i][s],f[i-1][s]);
        }
    }
    printf("%d\n", f[n][(1<<m)-1]);
    return 0;
}
