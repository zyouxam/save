#include<bits/stdc++.h>
using namespace std;
const int maxn=15e4+10;
long long dp[2][maxn],a,b,t,last_t=1,delta,q,m,d,n,line=0;;
int main(){
	scanf("%lld%lld%lld",&n,&m,&d); 
	while(m--){
		scanf("%lld%lld%lld",&a,&b,&t);
		delta=(t-last_t)*d, last_t=t;
		deque<int> q;
		for(int i=1,j=1;i<=n;i++){
			for(;j<=i+delta&&j<=n;j++){
				while(!q.empty()&&dp[line][q.back()]<=dp[line][j])q.pop_back();
				q.push_back(j);
			}
			while(!q.empty()&&q.front()<i-delta)q.pop_front();
			dp[!line][i]=dp[line][q.front()]+b-abs(i-a);
		}
		line=!line;
	}
	long long ans=dp[line][1];
	for(int i=2;i<=n;i++){
		if(dp[line][i]>ans)ans=dp[line][i];
	}
	printf("%lld",ans);
}
