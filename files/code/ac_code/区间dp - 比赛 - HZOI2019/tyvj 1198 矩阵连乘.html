#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
int n;
long long a[201], dp[201][201];
int main() {
    cin >> n;
    for (int i = 0; i <= n; i++) cin >> a[i];
    memset(dp, 0x7f, sizeof(dp));
    for (int i = 0; i <= n + 1; i++) dp[i][i] = 0;
    for (int len = 1; len <= n; len++) {    
        for (int i = 1; i <= n; i++) {
            int j = min(n, len + i - 1);
            for (int k = i; k < j; k++) {
                dp[i][j] = min(dp[i][j], dp[i][k] + dp[k + 1][j] + a[i - 1] * a[k] * a[j]);
            }
        }
    }
    cout << dp[1][n];
}
