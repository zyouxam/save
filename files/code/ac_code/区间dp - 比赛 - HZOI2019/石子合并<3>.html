#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
const int maxn = 4010, INF = 0x3f3f3f3f;
int f[maxn][maxn], sum[maxn], a[maxn];
int main() {
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        a[n + i] = a[i];
    }
    for (int i = 1; i <= n * 2; i++) {
        sum[i] = sum[i - 1] + a[i];
    }
    int ans = -INF;
    for (int d = 2; d <= n; d++) {
        for (int i = 1; i <= 2 * n - d + 1; i++) {
            int k = d + i - 1;
            f[i][k] = max(f[i + 1][k], f[i][k - 1]) + sum[k] - sum[i - 1];
        }
    }
    for (int i = 1; i <= n; i++) {
        ans = max(ans, f[i][i + n - 1]);
    }
    printf("%d\n", ans);
    return 0;
}
