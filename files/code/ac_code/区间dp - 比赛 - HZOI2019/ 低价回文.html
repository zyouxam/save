#include<bits/stdc++.h>
#define inf 1e8
using namespace std;
int dp[2020][2020];
int main() {
    int n,m,x,y,v[30];
    char z,a[2020];
    scanf("%d%d",&n,&m);
    scanf("%s",a);
    for(int i=0;i<n;i++){
        cin>>z>>x>>y;
        v[z-'a']=min(x,y);
    }
    for(int j=1;j<m;j++){
        for(int i=j-1;i>=0;i--){
            dp[i][j]=inf;
            if(a[i]==a[j]) dp[i][j]=dp[i+1][j-1];
            else dp[i][j]=min(dp[i+1][j]+v[a[i]-'a'],dp[i][j-1]+v[a[j]-'a']);
        }
    }
    printf("%d\n",dp[0][m-1]);
    return 0;
}
