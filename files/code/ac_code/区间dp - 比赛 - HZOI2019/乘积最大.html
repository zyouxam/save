#include <algorithm>
#include <cstdio>
#include <cstring>
using namespace std;
int main() {
    int n, k;
    char str[22];
    int mul[42][42], dp[42][7], cnt;
    while (scanf("%d %d", &n, &k) != EOF) {
        memset(dp, -1, sizeof(dp));
        memset(mul, 0, sizeof(mul));
        scanf("%s", str + 1);
        for (int i = 1; i <= n; i++) {
            for (int j = i; j <= n; j++) {
                cnt = 0;
                for (int m = i; m <= j; m++) cnt = cnt * 10 + str[m] - '0';
                mul[i][j] = cnt;
            }
        }
        for (int i = 1; i <= n; i++) dp[i][0] = mul[1][i];
        for (int j = 1; j <= k; j++)
            for (int i = j + 1; i <= n; i++)
                for (int m = j; m <= i; m++)
                    dp[i][j] = max(dp[i][j], dp[m][j - 1] * mul[m + 1][i]);
        printf("%d\n", dp[n][k]);
    }
}
