#include <iostream>
using namespace std;
int main() {
    int n;
    cin >> n;
    int a[2005] = {0};
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        a[i + n] = a[i];
    }
    int dp[200][200] = {0};
    int maxnum = 0;
    for (int i = 3; i <= 2 * n; i++) {
        for (int j = i - 2; j >= max(1, i - n); j--) {
            for (int k = j + 1; k < i; k++) {
                dp[j][i] =
                    max(dp[j][i], dp[j][k] + dp[k][i] + a[j] * a[k] * a[i]);
                maxnum = max(maxnum, dp[j][i]);
            }
        }
    }
    cout << maxnum << endl;
    return 0;
}
